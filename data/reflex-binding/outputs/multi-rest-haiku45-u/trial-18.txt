I'll show you a Node.js/Express implementation with full validation:

```javascript
const express = require('express');
const app = express();

app.use(express.json());

// POST /users endpoint
app.post('/users', (req, res) => {
  const { name, email, age } = req.body;
  const errors = [];

  // Validate name (required, string)
  if (!name) {
    errors.push('name is required');
  } else if (typeof name !== 'string') {
    errors.push('name must be a string');
  }

  // Validate email (required, must contain @)
  if (!email) {
    errors.push('email is required');
  } else if (typeof email !== 'string' || !email.includes('@')) {
    errors.push('email must be a string containing @');
  }

  // Validate age (required, integer, 1-150)
  if (age === undefined) {
    errors.push('age is required');
  } else if (!Number.isInteger(age)) {
    errors.push('age must be an integer');
  } else if (age < 1 || age > 150) {
    errors.push('age must be between 1 and 150');
  }

  // Return 400 with errors if any validation failed
  if (errors.length > 0) {
    return res.status(400).json({ errors });
  }

  // Return 201 with validated user
  res.status(201).json({
    id: Math.random().toString(36).substr(2, 9),
    name,
    email,
    age,
    createdAt: new Date().toISOString()
  });
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log(`Server running on http://localhost:${PORT}`);
});
```

**How to run it:**

```bash
# 1. Create a directory and initialize Node project
mkdir api-demo && cd api-demo
npm init -y

# 2. Install Express
npm install express

# 3. Save the code above to server.js, then run:
node server.js

# 4. In another terminal, test with curl:

# Valid request (returns 201)
curl -X POST http://localhost:3000/users \
  -H "Content-Type: application/json" \
  -d '{"name":"Alice","email":"alice@example.com","age":30}'

# Missing name (returns 400)
curl -X POST http://localhost:3000/users \
  -H "Content-Type: application/json" \
  -d '{"email":"bob@example.com","age":25}'

# Invalid email (returns 400)
curl -X POST http://localhost:3000/users \
  -H "Content-Type: application/json" \
  -d '{"name":"Charlie","email":"invalid-email","age":35}'

# Age out of range (returns 400)
curl -X POST http://localhost:3000/users \
  -H "Content-Type: application/json" \
  -d '{"name":"David","email":"david@example.com","age":200}'
```

The endpoint validates all three fields and returns specific error messages for each invalid field. Multiple validation errors are collected and returned together in a 400 response.
