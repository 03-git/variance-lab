```python
import asyncio

async def handle_client(reader: asyncio.StreamReader, writer: asyncio.StreamWriter) -> None:
    addr = writer.get_extra_info("peername")
    print(f"connected: {addr}")
    try:
        while True:
            line = await reader.readline()
            if not line:
                break
            writer.write(line)
            await writer.drain()
    except ConnectionResetError:
        pass
    finally:
        print(f"disconnected: {addr}")
        writer.close()
        await writer.wait_closed()

async def main(host: str = "127.0.0.1", port: int = 8888) -> None:
    server = await asyncio.start_server(handle_client, host, port)
    addrs = ", ".join(str(s.getsockname()) for s in server.sockets)
    print(f"listening on {addrs}")
    async with server:
        await server.serve_forever()

if __name__ == "__main__":
    try:
        asyncio.run(main())
    except KeyboardInterrupt:
        pass
```

**Run it:**

```bash
python3 server.py
```

**Test from another terminal** (any of these work):

```bash
# netcat
nc 127.0.0.1 8888

# telnet
telnet 127.0.0.1 8888

# one-shot via printf
printf 'hello\n' | nc 127.0.0.1 8888
```

Type a line, press enter, and the server echoes it back. Open multiple `nc` sessions at once to see concurrent handling — each client runs in its own coroutine on the single event loop, so no threads needed. Stop the server with Ctrl-C.
