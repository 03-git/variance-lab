```python
import urllib.request
from html.parser import HTMLParser

class LinkParser(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []
        self._current_href = None
        self._capture = False

    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            attrs = dict(attrs)
            self._current_href = attrs.get('href')
            self._capture = True

    def handle_endtag(self, tag):
        if tag == 'a':
            self._capture = False
            self._current_href = None

    def handle_data(self, data):
        if self._capture and self._current_href:
            self.links.append((self._current_href, data.strip()))

with urllib.request.urlopen('https://example.com') as resp:
    html = resp.read().decode('utf-8')

parser = LinkParser()
parser.feed(html)

for href, text in parser.links:
    print(f"{text!r:30s} -> {href}")
```

**Run it:**

```sh
python3 script.py
```

No dependencies — uses only the stdlib `urllib` and `html.parser`. The `HTMLParser` is stateful: it records the `href` on `<a>` open, captures text between tags, and clears state on `</a>`.
