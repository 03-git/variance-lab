```python
import urllib.request
from html.parser import HTMLParser

class LinkParser(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []
        self._current_href = None
        self._capturing = False
        self._text_parts = []

    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            attrs_dict = dict(attrs)
            self._current_href = attrs_dict.get('href')
            self._capturing = True
            self._text_parts = []

    def handle_endtag(self, tag):
        if tag == 'a' and self._capturing:
            self.links.append((self._current_href, ''.join(self._text_parts).strip()))
            self._capturing = False
            self._current_href = None
            self._text_parts = []

    def handle_data(self, data):
        if self._capturing:
            self._text_parts.append(data)

with urllib.request.urlopen('https://example.com') as response:
    html = response.read().decode('utf-8')

parser = LinkParser()
parser.feed(html)

for href, text in parser.links:
    print(f"{text!r:30} -> {href}")
```

**To run:**

```bash
python3 script.py
```

No dependencies beyond the standard library — `urllib.request` and `html.parser` are both built in. For a site with many links you'd see one line per anchor; `example.com` itself has one link (`More information...` -> `https://www.iana.org/domains/reserved`).
