```python
import requests
from html.parser import HTMLParser

class LinkExtractor(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []
        self.current_text = []
    
    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            attrs_dict = dict(attrs)
            href = attrs_dict.get('href', '')
            self.current_text = []
    
    def handle_data(self, data):
        self.current_text.append(data.strip())
    
    def handle_endtag(self, tag):
        if tag == 'a':
            text = ' '.join(self.current_text).strip()
            if hasattr(self, '_current_href'):
                self.links.append({
                    'href': self._current_href,
                    'text': text
                })
            self.current_text = []
    
    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            attrs_dict = dict(attrs)
            self._current_href = attrs_dict.get('href', '')

def fetch_and_extract_links(url):
    try:
        response = requests.get(url, timeout=10)
        response.raise_for_status()
        
        parser = LinkExtractor()
        parser.feed(response.text)
        
        return parser.links
    except requests.RequestException as e:
        print(f"Error fetching {url}: {e}")
        return []

if __name__ == '__main__':
    url = 'https://example.com'
    links = fetch_and_extract_links(url)
    
    if links:
        print(f"Found {len(links)} links:\n")
        for link in links:
            href = link['href']
            text = link['text']
            print(f"  href: {href}")
            print(f"  text: {text}")
            print()
    else:
        print("No links found or error occurred.")
```

**What it does:**
- Uses `requests` to fetch the page
- Custom `HTMLParser` extracts all `<a>` tags
- For each link, captures the `href` attribute and the link text
- Prints each href and its corresponding link text

**Note:** The custom parser has a small issue in my first version. Here's the corrected version:

```python
import requests
from html.parser import HTMLParser

class LinkExtractor(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []
        self.current_href = None
        self.current_text = []
    
    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            attrs_dict = dict(attrs)
            self.current_href = attrs_dict.get('href', '')
            self.current_text = []
    
    def handle_data(self, data):
        if self.current_href is not None:
            self.current_text.append(data.strip())
    
    def handle_endtag(self, tag):
        if tag == 'a' and self.current_href is not None:
            text = ' '.join(self.current_text).strip()
            self.links.append({'href': self.current_href, 'text': text})
            self.current_href = None
            self.current_text = []

def fetch_and_extract_links(url):
    try:
        response = requests.get(url, timeout=10)
        response.raise_for_status()
        parser = LinkExtractor()
        parser.feed(response.text)
        return parser.links
    except requests.RequestException as e:
        print(f"Error fetching {url}: {e}")
        return []

if __name__ == '__main__':
    url = 'https://example.com'
    links = fetch_and_extract_links(url)
    
    if links:
        print(f"Found {len(links)} links:\n")
        for link in links:
            print(f"  href: {link['href']}")
            print(f"  text: {link['text']}")
            print()
    else:
        print("No links found or error occurred.")
```

This corrected version properly tracks the href while collecting the text content between the opening and closing `</a>` tags.
