```python
import urllib.request
from html.parser import HTMLParser

class LinkExtractor(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []
        self._href = None
        self._text = ""

    def handle_starttag(self, tag, attrs):
        if tag == "a":
            for name, value in attrs:
                if name == "href":
                    self._href = value
                    self._text = ""

    def handle_data(self, data):
        if self._href is not None:
            self._text += data

    def handle_endtag(self, tag):
        if tag == "a" and self._href is not None:
            self.links.append((self._href, self._text.strip()))
            self._href = None
            self._text = ""

html = urllib.request.urlopen("https://example.com").read().decode()
parser = LinkExtractor()
parser.feed(html)

for href, text in parser.links:
    print(f"{href}  ->  {text}")
```

Run it:

```
python3 script.py
```

No dependencies beyond the standard library — `urllib.request` for the fetch, `html.parser.HTMLParser` for extraction.
