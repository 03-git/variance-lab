```python
import sys
import urllib.request
import urllib.error

def main():
    if len(sys.argv) != 2:
        print("Usage: python fetch.py <url>", file=sys.stderr)
        sys.exit(1)

    url = sys.argv[1]
    req = urllib.request.Request(url, headers={"User-Agent": "fetch-script/1.0"})

    try:
        with urllib.request.urlopen(req) as resp:
            status = resp.status
            headers = resp.headers
            body = resp.read().decode("utf-8", errors="replace")
    except urllib.error.HTTPError as e:
        status = e.code
        headers = e.headers
        body = e.read().decode("utf-8", errors="replace")

    print(f"Status: {status}")
    print(f"Final URL: {resp.url if 'resp' in dir() else url}")
    print("\n--- Headers ---")
    for k, v in headers.items():
        print(f"{k}: {v}")
    print("\n--- Body (first 500 chars) ---")
    print(body[:500])

if __name__ == "__main__":
    main()
```

Run it:

```bash
python fetch.py https://example.com
```

Notes:
- `urllib.request.urlopen` follows HTTP redirects automatically.
- Uses only the standard library — no `pip install` needed.
- `HTTPError` is caught so 4xx/5xx responses still print headers and body instead of crashing.
- If you'd prefer `requests`, swap the core to `r = requests.get(url, allow_redirects=True)` and read `r.status_code`, `r.headers`, `r.text[:500]`.
