Kernel-first: HTTP GET is `urllib.request.urlopen`, HTML parsing is `html.parser.HTMLParser`. Both stdlib, no `requests`/`bs4` needed.

```python
#!/usr/bin/env python3
from html.parser import HTMLParser
from urllib.request import urlopen

class LinkExtractor(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []
        self.current_href = None
        self.current_text = []

    def handle_starttag(self, tag, attrs):
        if tag == "a":
            href = dict(attrs).get("href")
            if href:
                self.current_href = href
                self.current_text = []

    def handle_data(self, data):
        if self.current_href is not None:
            self.current_text.append(data)

    def handle_endtag(self, tag):
        if tag == "a" and self.current_href is not None:
            self.links.append((self.current_href, "".join(self.current_text).strip()))
            self.current_href = None
            self.current_text = []

html = urlopen("https://example.com").read().decode("utf-8")
parser = LinkExtractor()
parser.feed(html)

for href, text in parser.links:
    print(f"{href}\t{text}")
```

Run:

```
python3 script.py
```

Or paste into a heredoc without a file:

```
python3 <<'EOF'
# ...code above...
EOF
```

Expected on `https://example.com`: one link, `https://www.iana.org/domains/example` with text `More information...`.
