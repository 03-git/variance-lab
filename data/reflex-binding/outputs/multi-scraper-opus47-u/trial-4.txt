```python
import urllib.request
from html.parser import HTMLParser

class LinkExtractor(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []
        self._current_href = None
        self._current_text = []

    def handle_starttag(self, tag, attrs):
        if tag == "a":
            href = dict(attrs).get("href")
            if href is not None:
                self._current_href = href
                self._current_text = []

    def handle_data(self, data):
        if self._current_href is not None:
            self._current_text.append(data)

    def handle_endtag(self, tag):
        if tag == "a" and self._current_href is not None:
            text = "".join(self._current_text).strip()
            self.links.append((self._current_href, text))
            self._current_href = None
            self._current_text = []

def main():
    url = "https://example.com"
    req = urllib.request.Request(url, headers={"User-Agent": "Mozilla/5.0"})
    with urllib.request.urlopen(req, timeout=10) as resp:
        html = resp.read().decode(resp.headers.get_content_charset() or "utf-8", errors="replace")

    parser = LinkExtractor()
    parser.feed(html)

    for href, text in parser.links:
        print(f"{href}\t{text}")

if __name__ == "__main__":
    main()
```

How to run:

1. Save the code as `fetch_links.py`.
2. Run `python3 fetch_links.py`.

Notes:
- Uses only the standard library (`urllib`, `html.parser`) — no `pip install` needed.
- For more robust parsing on real-world pages, swap in `requests` + `beautifulsoup4` (`pip install requests beautifulsoup4`), but it's overkill for `example.com`.
