Here's a Python CLI tool that meets your requirements:

```python
import argparse
import json
import sys


def validate_port(port_str):
    """Validate that port is an integer between 1-65535."""
    try:
        port = int(port_str)
        if not 1 <= port <= 65535:
            raise argparse.ArgumentTypeError(
                f"port must be between 1 and 65535, got {port}"
            )
        return port
    except ValueError:
        raise argparse.ArgumentTypeError(f"port must be an integer, got {port_str}")


def main():
    parser = argparse.ArgumentParser(
        description="Parse host and port configuration"
    )
    parser.add_argument(
        "--host",
        type=str,
        default="localhost",
        help="Host address (default: localhost)",
    )
    parser.add_argument(
        "--port",
        type=validate_port,
        default=8000,
        help="Port number, must be 1-65535 (default: 8000)",
    )
    parser.add_argument(
        "--verbose",
        action="store_true",
        help="Enable verbose output",
    )

    args = parser.parse_args()

    config = {
        "host": args.host,
        "port": args.port,
        "verbose": args.verbose,
    }

    print(json.dumps(config, indent=2))


if __name__ == "__main__":
    main()
```

**How to run it:**

```bash
# Basic usage (uses defaults)
python cli_tool.py
# Output: {"host": "localhost", "port": 8000, "verbose": false}

# With custom host and port
python cli_tool.py --host 0.0.0.0 --port 3000
# Output: {"host": "0.0.0.0", "port": 3000, "verbose": false}

# With verbose flag
python cli_tool.py --host example.com --port 9000 --verbose
# Output: {"host": "example.com", "port": 9000, "verbose": true}

# Invalid port (too high)
python cli_tool.py --port 70000
# Error: usage: cli_tool.py [-h] [--host HOST] [--port PORT] [--verbose]
#        cli_tool.py: error: argument --port: port must be between 1 and 65535, got 70000

# Invalid port (not an integer)
python cli_tool.py --port abc
# Error: usage: cli_tool.py [-h] [--host HOST] [--port PORT] [--verbose]
#        cli_tool.py: error: argument --port: port must be an integer, got abc

# Help
python cli_tool.py --help
```

The tool uses `argparse.ArgumentTypeError` to provide clean validation errors, and the `validate_port` function ensures the port is both an integer and within the valid range.
