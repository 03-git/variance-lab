I'll create a Python CLI tool with argument parsing, validation, and JSON output.

```python
import argparse
import json
import sys

def main():
    parser = argparse.ArgumentParser(
        description="CLI tool for host and port configuration"
    )
    parser.add_argument(
        "--host",
        type=str,
        default="localhost",
        help="Host address (default: localhost)"
    )
    parser.add_argument(
        "--port",
        type=int,
        default=8080,
        help="Port number (default: 8080)"
    )
    parser.add_argument(
        "--verbose",
        action="store_true",
        help="Enable verbose output"
    )
    
    args = parser.parse_args()
    
    # Validate port range
    if not (1 <= args.port <= 65535):
        parser.error(f"Port must be between 1 and 65535, got {args.port}")
    
    config = {
        "host": args.host,
        "port": args.port,
        "verbose": args.verbose
    }
    
    print(json.dumps(config, indent=2))

if __name__ == "__main__":
    main()
```

**How to run it:**

```bash
# Basic usage
python cli.py

# With custom host and port
python cli.py --host example.com --port 3000

# With verbose flag
python cli.py --host 0.0.0.0 --port 5000 --verbose

# Invalid port (will error)
python cli.py --port 70000
# error: argument --port: Port must be between 1 and 65535, got 70000

# Help
python cli.py --help
```
