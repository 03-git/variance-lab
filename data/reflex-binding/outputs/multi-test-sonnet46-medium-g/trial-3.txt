```python
import unittest

def celsius_to_fahrenheit(c):
    return c * 9/5 + 32

class TestCelsiusToFahrenheit(unittest.TestCase):

    def test_freezing(self):
        self.assertAlmostEqual(celsius_to_fahrenheit(0), 32.0)

    def test_boiling(self):
        self.assertAlmostEqual(celsius_to_fahrenheit(100), 212.0)

    def test_negative_forty(self):
        # -40 is where Celsius and Fahrenheit converge
        self.assertAlmostEqual(celsius_to_fahrenheit(-40), -40.0)

    def test_non_numeric_raises(self):
        with self.assertRaises(TypeError):
            celsius_to_fahrenheit("hot")

if __name__ == "__main__":
    unittest.main()
```

**To run it**, save as e.g. `test_temp.py` and:

```bash
python test_temp.py -v
```
