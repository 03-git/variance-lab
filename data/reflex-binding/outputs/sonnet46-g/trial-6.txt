Here's a minimal implementation in a few language options — pick what fits your stack:

**Python (stdlib only)**
```python
from http.server import HTTPServer, BaseHTTPRequestHandler
from datetime import datetime, timezone
import json

class Handler(BaseHTTPRequestHandler):
    def do_GET(self):
        if self.path == '/time':
            body = json.dumps({"time": datetime.now(timezone.utc).isoformat()}).encode()
            self.send_response(200)
            self.send_header('Content-Type', 'application/json')
            self.end_headers()
            self.wfile.write(body)
        else:
            self.send_response(404)
            self.end_headers()
    def log_message(self, *_): pass

HTTPServer(('', 8080), Handler).serve_forever()
```
```
python3 time_server.py
curl http://localhost:8080/time
```

**Go (stdlib only)**
```go
package main

import (
    "encoding/json"
    "net/http"
    "time"
)

func main() {
    http.HandleFunc("/time", func(w http.ResponseWriter, r *http.Request) {
        w.Header().Set("Content-Type", "application/json")
        json.NewEncoder(w).Encode(map[string]string{"time": time.Now().UTC().Format(time.RFC3339)})
    })
    http.ListenAndServe(":8080", nil)
}
```
```
go run main.go
curl http://localhost:8080/time
```

**C (POSIX + stdio)**
```c
#include <stdio.h>
#include <string.h>
#include <time.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <unistd.h>

int main(void) {
    int srv = socket(AF_INET, SOCK_STREAM, 0);
    int opt = 1;
    setsockopt(srv, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    struct sockaddr_in addr = {AF_INET, htons(8080), {INADDR_ANY}};
    bind(srv, (struct sockaddr*)&addr, sizeof(addr));
    listen(srv, 8);

    for (;;) {
        int fd = accept(srv, NULL, NULL);
        char req[1024];
        read(fd, req, sizeof(req));

        time_t t = time(NULL);
        struct tm *tm = gmtime(&t);
        char ts[32];
        strftime(ts, sizeof(ts), "%Y-%m-%dT%H:%M:%SZ", tm);

        char body[128], resp[512];
        snprintf(body, sizeof(body), "{\"time\":\"%s\"}", ts);
        snprintf(resp, sizeof(resp),
            "HTTP/1.1 200 OK\r\nContent-Type: application/json\r\nContent-Length: %zu\r\n\r\n%s",
            strlen(body), body);

        write(fd, resp, strlen(resp));
        close(fd);
    }
}
```
```
gcc -o time_server time_server.c && ./time_server
curl http://localhost:8080/time
```

---

All three return something like `{"time":"2026-05-10T14:32:01Z"}`. The Go and C versions use `RFC3339`/`strftime` which give second-precision UTC; the Python version uses `isoformat()` which includes microseconds — strip with `.replace(microsecond=0)` if you want clean seconds.
