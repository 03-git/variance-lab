Here's a minimal implementation using Python's standard library — no dependencies, no files needed:

```python
# server.py
from http.server import BaseHTTPRequestHandler, HTTPServer
from datetime import datetime, timezone
import json

class Handler(BaseHTTPRequestHandler):
    def do_GET(self):
        if self.path != "/time":
            self.send_response(404)
            self.end_headers()
            return
        body = json.dumps({"time": datetime.now(timezone.utc).isoformat()}).encode()
        self.send_response(200)
        self.send_header("Content-Type", "application/json")
        self.send_header("Content-Length", str(len(body)))
        self.end_headers()
        self.wfile.write(body)

    def log_message(self, *args):  # quiet logging; remove to see requests
        pass

if __name__ == "__main__":
    HTTPServer(("127.0.0.1", 8000), Handler).serve_forever()
```

Run it:

```sh
python3 server.py
```

Test it:

```sh
curl http://127.0.0.1:8000/time
# {"time": "2026-05-10T18:42:11.123456+00:00"}
```

```sh
python3 -c '
from http.server import BaseHTTPRequestHandler, HTTPServer
from datetime import datetime, timezone
import json
class H(BaseHTTPRequestHandler):
    def do_GET(self):
        b = json.dumps({"time": datetime.now(timezone.utc).isoformat()}).encode()
        self.send_response(200 if self.path=="/time" else 404)
        self.send_header("Content-Type","application/json"); self.end_headers()
        if self.path=="/time": self.wfile.write(b)
HTTPServer(("127.0.0.1",8000),H).serve_forever()'
```

Notes:
- `datetime.now(timezone.utc).isoformat()` produces an ISO 8601 string like `2026-05-10T18:42:11.123456+00:00`. If you want the `Z` suffix style instead, use `datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")`.
