Here's a Python script that fetches a URL and extracts hyperlinks:

```python
import requests
from html.parser import HTMLParser

class LinkExtractor(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []
        self.current_text = []
    
    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            href = dict(attrs).get('href', '')
            self.current_text = []
            self.links.append({'href': href, 'text': ''})
    
    def handle_data(self, data):
        if self.links:
            self.current_text.append(data)
    
    def handle_endtag(self, tag):
        if tag == 'a' and self.links:
            self.links[-1]['text'] = ''.join(self.current_text).strip()

def extract_links(url):
    try:
        response = requests.get(url, timeout=10)
        response.raise_for_status()
        
        parser = LinkExtractor()
        parser.feed(response.text)
        
        for link in parser.links:
            print(f"href: {link['href']}")
            print(f"text: {link['text']}")
            print()
    
    except requests.RequestException as e:
        print(f"Error fetching {url}: {e}")

if __name__ == '__main__':
    extract_links('https://example.com')
```

**To run it:**

```bash
# Install requests if you don't have it
pip install requests

# Run the script
python script.py
```

The script uses the standard `HTMLParser` to parse HTML (no external dependencies except `requests`) and extracts all `<a>` tags along with their `href` attributes and link text.
