I'll show you a Python script that fetches a URL and extracts hyperlinks.

```python
import requests
from html.parser import HTMLParser

class LinkExtractor(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []
        self.current_text = []
    
    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            href = dict(attrs).get('href', '')
            self.current_text = []
    
    def handle_data(self, data):
        self.current_text.append(data.strip())
    
    def handle_endtag(self, tag):
        if tag == 'a':
            text = ' '.join(self.current_text).strip()
            self.links.append((text, dict(getattr(self, '_attrs', [])) if hasattr(self, '_attrs') else ''))
    
    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            self._attrs = attrs
            self.current_text = []
    
    def handle_endtag(self, tag):
        if tag == 'a':
            href = dict(self._attrs).get('href', '')
            text = ' '.join(self.current_text).strip()
            self.links.append((href, text))

def extract_links(url):
    try:
        response = requests.get(url, timeout=10)
        response.raise_for_status()
        
        parser = LinkExtractor()
        parser.feed(response.text)
        
        return parser.links
    except requests.RequestException as e:
        print(f"Error fetching {url}: {e}")
        return []

if __name__ == "__main__":
    url = "https://example.com"
    links = extract_links(url)
    
    print(f"Found {len(links)} links:\n")
    for href, text in links:
        print(f"href: {href}")
        print(f"text: {text}\n")
```

**To run it:**

1. Install the required library:
```bash
pip install requests
```

2. Run the script:
```bash
python script.py
```

The script uses the `requests` library to fetch the HTML and a custom `HTMLParser` to extract all `<a>` tags with their `href` attributes and link text. Each link is printed with its URL and the visible text.
