I'll show you a Python script that fetches a URL and extracts all hyperlinks from the HTML.

```python
import requests
from html.parser import HTMLParser
from urllib.parse import urljoin

class LinkExtractor(HTMLParser):
    def __init__(self, base_url):
        super().__init__()
        self.base_url = base_url
        self.links = []
        self.current_text = []
    
    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            href = next((value for attr, value in attrs if attr == 'href'), None)
            if href:
                # Convert relative URLs to absolute
                absolute_url = urljoin(self.base_url, href)
                self.links.append({'href': absolute_url, 'text': ''})
    
    def handle_data(self, data):
        if self.links:
            self.links[-1]['text'] += data.strip()
    
    def get_links(self):
        return self.links

def fetch_and_extract_links(url):
    try:
        response = requests.get(url, timeout=10)
        response.raise_for_status()
        
        parser = LinkExtractor(url)
        parser.feed(response.text)
        
        return parser.get_links()
    except requests.RequestException as e:
        print(f"Error fetching {url}: {e}")
        return []

if __name__ == '__main__':
    url = 'https://example.com'
    links = fetch_and_extract_links(url)
    
    print(f"Found {len(links)} links:\n")
    for link in links:
        href = link['href']
        text = link['text'] or '(no text)'
        print(f"Text: {text}")
        print(f"URL:  {href}\n")
```

**To run it:**

1. Install the required library:
   ```bash
   pip install requests
   ```

2. Run the script:
   ```bash
   python script.py
   ```

The script:
- Fetches the HTML from the URL
- Uses `HTMLParser` to parse anchor tags and extract `href` attributes and link text
- Converts relative URLs to absolute URLs using `urljoin`
- Prints each link's text and URL

You can modify the `url` variable to fetch from any website.
