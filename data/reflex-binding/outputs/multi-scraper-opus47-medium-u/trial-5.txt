```python
#!/usr/bin/env python3
"""Fetch a URL and print every <a> tag's href and link text."""

import sys
import urllib.request
from html.parser import HTMLParser
from urllib.parse import urljoin


class LinkParser(HTMLParser):
    def __init__(self, base_url):
        super().__init__()
        self.base_url = base_url
        self.links = []          # list of (href, text)
        self._cur_href = None
        self._cur_text = []

    def handle_starttag(self, tag, attrs):
        if tag == "a":
            href = dict(attrs).get("href")
            if href is not None:
                self._cur_href = urljoin(self.base_url, href)
                self._cur_text = []

    def handle_data(self, data):
        if self._cur_href is not None:
            self._cur_text.append(data)

    def handle_endtag(self, tag):
        if tag == "a" and self._cur_href is not None:
            text = " ".join("".join(self._cur_text).split())
            self.links.append((self._cur_href, text))
            self._cur_href = None
            self._cur_text = []


def main():
    url = sys.argv[1] if len(sys.argv) > 1 else "https://example.com"
    req = urllib.request.Request(url, headers={"User-Agent": "link-extractor/1.0"})
    with urllib.request.urlopen(req, timeout=10) as resp:
        charset = resp.headers.get_content_charset() or "utf-8"
        html = resp.read().decode(charset, errors="replace")

    parser = LinkParser(url)
    parser.feed(html)

    if not parser.links:
        print("No links found.")
        return
    for href, text in parser.links:
        print(f"{href}\t{text or '(no text)'}")


if __name__ == "__main__":
    main()
```

How to run it:

```bash
# Save the code as link_extractor.py, then:
python3 link_extractor.py

# Or pass any other URL:
python3 link_extractor.py https://www.python.org
```

Notes:
- Uses only the Python standard library — no `pip install` needed.
- `urljoin` turns relative hrefs (like `/about`) into absolute URLs.
- Output is `href<TAB>link text`, one link per line. `https://example.com` itself has just one link ("More information..." → iana.org).
- For heavier scraping you'd typically use `requests` + `beautifulsoup4`, but that's overkill here.
