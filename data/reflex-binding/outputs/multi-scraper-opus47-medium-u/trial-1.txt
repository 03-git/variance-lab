```python
#!/usr/bin/env python3
"""Fetch a URL and print every <a> tag's href and link text."""

import sys
from html.parser import HTMLParser
from urllib.request import urlopen, Request


class LinkParser(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []          # list of (href, text)
        self._href = None        # href of the anchor currently open
        self._text = []          # text chunks collected inside it

    def handle_starttag(self, tag, attrs):
        if tag == "a":
            href = dict(attrs).get("href")
            if href is not None:
                self._href = href
                self._text = []

    def handle_data(self, data):
        if self._href is not None:
            self._text.append(data)

    def handle_endtag(self, tag):
        if tag == "a" and self._href is not None:
            text = " ".join("".join(self._text).split())
            self.links.append((self._href, text))
            self._href = None
            self._text = []


def main():
    url = sys.argv[1] if len(sys.argv) > 1 else "https://example.com"
    req = Request(url, headers={"User-Agent": "link-extractor/1.0"})
    with urlopen(req) as resp:
        charset = resp.headers.get_content_charset() or "utf-8"
        html = resp.read().decode(charset, errors="replace")

    parser = LinkParser()
    parser.feed(html)

    if not parser.links:
        print("No links found.")
        return
    for href, text in parser.links:
        print(f"{href}\t{text!r}")


if __name__ == "__main__":
    main()
```

How to run it:

1. Save it as `extract_links.py` (you said no files — so paste it into your own editor).
2. Run with the stdlib only (no `pip install` needed), Python 3.x:

```bash
python3 extract_links.py
# or pass a different URL:
python3 extract_links.py https://www.python.org
```

Expected output for `https://example.com` (it has a single link):

```
https://www.iana.org/domains/example	'More information...'
```

Notes:
- Uses only the standard library (`urllib`, `html.parser`) so it runs anywhere Python 3 is installed.
- Output is `href<TAB>'link text'`. Relative hrefs are printed as-is; wrap with `urllib.parse.urljoin(url, href)` if you want them absolute.
- For heavier real-world scraping, `requests` + `beautifulsoup4` is the common upgrade, but it's not needed here.
