```python
import sys
import urllib.request
import urllib.error

def fetch(url):
    req = urllib.request.Request(url, headers={"User-Agent": "fetch-script/1.0"})
    try:
        with urllib.request.urlopen(req) as resp:
            body = resp.read().decode("utf-8", errors="replace")
            print(f"Status: {resp.status}")
            print("Headers:")
            for k, v in resp.headers.items():
                print(f"  {k}: {v}")
            print("\nBody (first 500 chars):")
            print(body[:500])
    except urllib.error.HTTPError as e:
        body = e.read().decode("utf-8", errors="replace")
        print(f"Status: {e.code}")
        print("Headers:")
        for k, v in e.headers.items():
            print(f"  {k}: {v}")
        print("\nBody (first 500 chars):")
        print(body[:500])

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("Usage: python fetch.py <url>", file=sys.stderr)
        sys.exit(1)
    fetch(sys.argv[1])
```

Run it:

```
python fetch.py https://example.com
```

`urllib.request.urlopen` follows HTTP redirects by default via `HTTPRedirectHandler`. Uses only the standard library — no `pip install` needed. If you'd prefer the `requests` library (cleaner API, same behavior with `allow_redirects=True`), let me know and I'll swap it.
