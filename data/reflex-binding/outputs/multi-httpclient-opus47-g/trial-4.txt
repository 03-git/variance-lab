```python
import sys
import urllib.request
import urllib.error

if len(sys.argv) != 2:
    print("Usage: python fetch.py <url>", file=sys.stderr)
    sys.exit(1)

url = sys.argv[1]
req = urllib.request.Request(url, headers={"User-Agent": "fetch.py/1.0"})

try:
    with urllib.request.urlopen(req) as resp:
        status = resp.status
        headers = resp.headers
        body = resp.read(500).decode(resp.headers.get_content_charset() or "utf-8", errors="replace")
except urllib.error.HTTPError as e:
    status = e.code
    headers = e.headers
    body = e.read(500).decode("utf-8", errors="replace")

print(f"Status: {status}\n")
print("Headers:")
for k, v in headers.items():
    print(f"  {k}: {v}")
print("\nBody (first 500 chars):")
print(body)
```

Run:
```
python fetch.py https://example.com
```

`urllib.request.urlopen` follows HTTP redirects by default. Stdlib only — no `pip install` needed.
