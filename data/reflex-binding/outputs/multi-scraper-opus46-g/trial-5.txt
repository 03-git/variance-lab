Stdlib only — no `requests`, no `bs4`. The kernel primitive under the hood is a TCP socket; `urllib.request` is the thinnest stdlib wrapper over it, and `html.parser` ships with Python.

```python
#!/usr/bin/env python3
"""Fetch a URL and print every <a> href with its link text."""
import sys
from urllib.request import urlopen, Request
from html.parser import HTMLParser
from urllib.parse import urljoin


class LinkExtractor(HTMLParser):
    def __init__(self, base_url):
        super().__init__()
        self.base_url = base_url
        self.links = []          # list of (href, text)
        self._href = None
        self._text_parts = []

    def handle_starttag(self, tag, attrs):
        if tag == "a":
            href = dict(attrs).get("href")
            if href is not None:
                self._href = urljoin(self.base_url, href)
                self._text_parts = []

    def handle_data(self, data):
        if self._href is not None:
            self._text_parts.append(data)

    def handle_endtag(self, tag):
        if tag == "a" and self._href is not None:
            text = " ".join("".join(self._text_parts).split())
            self.links.append((self._href, text))
            self._href = None
            self._text_parts = []


def main():
    url = sys.argv[1] if len(sys.argv) > 1 else "https://example.com"
    req = Request(url, headers={"User-Agent": "link-extractor/1.0"})
    with urlopen(req, timeout=10) as resp:
        charset = resp.headers.get_content_charset() or "utf-8"
        html = resp.read().decode(charset, errors="replace")

    parser = LinkExtractor(url)
    parser.feed(html)

    if not parser.links:
        print("(no anchor tags found)")
    for href, text in parser.links:
        print(f"{href}\t{text or '(no text)'}")


if __name__ == "__main__":
    main()
```

Run it:

```bash
python3 links.py                 # defaults to https://example.com
python3 links.py https://news.ycombinator.com   # or pass any URL
```

For `https://example.com` you'll get one line — its only link is to `https://www.iana.org/domains/example`:

```
https://www.iana.org/domains/example	More information...
```

Notes:
- `urljoin` resolves relative hrefs (`/about`, `../x`) against the page URL so output is always absolute.
- Link text is whitespace-collapsed; anchors wrapping images/empty content print `(no text)`.
- `html.parser` is lenient with real-world markup but won't execute JS — pages that build links client-side won't show up. If you ever need that, the primitive changes (headless browser), not the library.
