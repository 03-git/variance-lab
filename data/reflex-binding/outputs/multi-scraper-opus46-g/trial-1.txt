Using only the stdlib — the kernel primitive here is a TCP socket doing an HTTP GET; `urllib.request` is the thin wrapper over it, and `html.parser` (a C-backed state machine) is the parse primitive. No third-party deps needed.

```python
#!/usr/bin/env python3
"""Fetch a URL and print every <a> href and its link text."""
import sys
import urllib.request
from html.parser import HTMLParser
from urllib.parse import urljoin


class LinkExtractor(HTMLParser):
    def __init__(self, base_url):
        super().__init__()
        self.base_url = base_url
        self.links = []          # list of [href, text]
        self._in_a = False

    def handle_starttag(self, tag, attrs):
        if tag == "a":
            href = dict(attrs).get("href")
            if href is not None:
                self.links.append([urljoin(self.base_url, href), ""])
                self._in_a = True

    def handle_endtag(self, tag):
        if tag == "a":
            self._in_a = False

    def handle_data(self, data):
        if self._in_a and self.links:
            self.links[-1][1] += data.strip()


def main():
    url = sys.argv[1] if len(sys.argv) > 1 else "https://example.com"
    req = urllib.request.Request(url, headers={"User-Agent": "link-extractor/1.0"})
    with urllib.request.urlopen(req, timeout=10) as resp:
        charset = resp.headers.get_content_charset() or "utf-8"
        html = resp.read().decode(charset, errors="replace")

    parser = LinkExtractor(url)
    parser.feed(html)

    if not parser.links:
        print("(no <a> tags found)")
        return
    for href, text in parser.links:
        print(f"{href}\t{text or '(no text)'}")


if __name__ == "__main__":
    main()
```

How to run:

```bash
# default: fetches https://example.com
python3 link_extractor.py

# or pass any URL
python3 link_extractor.py https://example.com
```

Expected output for `example.com` (it has exactly one link):

```
https://www.iana.org/domains/example	More information...
```

Notes:
- `urljoin` resolves relative hrefs (`/about`, `../x`) against the page URL so you get absolute links.
- `html.parser` is lenient with real-world markup; for messy pages or JS-rendered links you'd need `lxml`/`beautifulsoup4` or a headless browser, but for static HTML this is enough.
- Add error handling around `urlopen` (catch `urllib.error.URLError`) if you'll run it on arbitrary/unreliable URLs.
