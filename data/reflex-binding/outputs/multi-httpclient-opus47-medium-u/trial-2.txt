```python
#!/usr/bin/env python3
"""Fetch a URL (following redirects) and print status, headers, and body preview."""

import sys
import urllib.request
import urllib.error


def fetch(url):
    req = urllib.request.Request(url, headers={"User-Agent": "url-fetcher/1.0"})
    try:
        # urllib follows redirects by default via HTTPRedirectHandler
        with urllib.request.urlopen(req) as resp:
            status = resp.status
            headers = resp.headers
            body = resp.read(500).decode(resp.headers.get_content_charset() or "utf-8", errors="replace")
    except urllib.error.HTTPError as e:
        # HTTPError is also a response-like object
        status = e.code
        headers = e.headers
        body = e.read(500).decode("utf-8", errors="replace")

    print(f"Status: {status}\n")
    print("Headers:")
    for k, v in headers.items():
        print(f"  {k}: {v}")
    print("\nBody (first 500 chars):")
    print(body)


if __name__ == "__main__":
    if len(sys.argv) != 2:
        print(f"Usage: {sys.argv[0]} <url>", file=sys.stderr)
        sys.exit(1)
    fetch(sys.argv[1])
```

If you'd rather use the `requests` library (cleaner, but a third-party dependency):

```python
#!/usr/bin/env python3
import sys
import requests

if len(sys.argv) != 2:
    sys.exit(f"Usage: {sys.argv[0]} <url>")

resp = requests.get(sys.argv[1], allow_redirects=True, timeout=30,
                    headers={"User-Agent": "url-fetcher/1.0"})

print(f"Status: {resp.status_code}\n")
print("Headers:")
for k, v in resp.headers.items():
    print(f"  {k}: {v}")
print("\nBody (first 500 chars):")
print(resp.text[:500])
```

**How to run:**

```bash
# Save either version as fetch_url.py, then:
python3 fetch_url.py https://example.com

# The stdlib version needs nothing extra. For the requests version:
pip install requests
```

The stdlib `urllib` version follows redirects automatically. Note that if the final response is an error status (4xx/5xx), it's caught via `HTTPError` so you still see the status, headers, and body. Want me to add anything — redirect chain tracing, JSON pretty-printing, timeout flag?
