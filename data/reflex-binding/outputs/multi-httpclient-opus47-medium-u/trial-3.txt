```python
#!/usr/bin/env python3
"""Fetch a URL and print status, headers, and a body preview."""
import sys
import urllib.request
import urllib.error


def main():
    if len(sys.argv) != 2:
        print(f"Usage: {sys.argv[0]} <url>", file=sys.stderr)
        sys.exit(1)

    url = sys.argv[1]
    req = urllib.request.Request(url, headers={"User-Agent": "fetch-script/1.0"})

    try:
        # urllib follows redirects (301/302/303/307) by default via HTTPRedirectHandler
        with urllib.request.urlopen(req) as resp:
            status = resp.status
            headers = resp.headers
            body = resp.read(500).decode("utf-8", errors="replace")
    except urllib.error.HTTPError as e:
        # HTTPError is also a response object — report it instead of crashing
        status = e.code
        headers = e.headers
        body = e.read(500).decode("utf-8", errors="replace")
    except urllib.error.URLError as e:
        print(f"Request failed: {e.reason}", file=sys.stderr)
        sys.exit(1)

    print(f"Status: {status}\n")
    print("Headers:")
    for key, value in headers.items():
        print(f"  {key}: {value}")
    print("\nBody (first 500 chars):")
    print(body)


if __name__ == "__main__":
    main()
```

**How to run it:**

```bash
# Save the code as fetch.py, then:
python3 fetch.py https://example.com

# or make it executable:
chmod +x fetch.py
./fetch.py https://httpbin.org/redirect/2
```

Notes:
- Uses only the standard library — no `pip install` needed.
- `urllib.request.urlopen` follows redirects automatically. The final `resp.url` would show the redirected-to URL if you want to print that too.
- `HTTPError` (4xx/5xx) is caught so you still see the status, headers, and body for error responses.
- If you'd rather use the `requests` library: `requests.get(url)` follows redirects by default (`r.status_code`, `r.headers`, `r.text[:500]`), but it's a third-party dependency.
