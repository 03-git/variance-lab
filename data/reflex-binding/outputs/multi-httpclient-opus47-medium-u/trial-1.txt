```python
#!/usr/bin/env python3
"""Fetch a URL and print status, headers, and a body preview."""

import sys
import urllib.request
import urllib.error


def fetch(url: str) -> None:
    req = urllib.request.Request(url, headers={"User-Agent": "url-fetch/1.0"})
    try:
        # urllib follows redirects (3xx) automatically via HTTPRedirectHandler
        with urllib.request.urlopen(req) as resp:
            status = resp.status
            headers = resp.headers
            body = resp.read(500).decode(resp.headers.get_content_charset() or "utf-8", errors="replace")
    except urllib.error.HTTPError as e:
        # HTTPError is also a response-like object
        status = e.code
        headers = e.headers
        body = e.read(500).decode("utf-8", errors="replace")

    print(f"Status: {status}\n")
    print("Headers:")
    for k, v in headers.items():
        print(f"  {k}: {v}")
    print("\nBody (first 500 chars):")
    print(body)


if __name__ == "__main__":
    if len(sys.argv) != 2:
        print(f"Usage: {sys.argv[0]} <url>", file=sys.stderr)
        sys.exit(1)
    fetch(sys.argv[1])
```

If you'd rather use the `requests` library (cleaner, but needs `pip install requests`):

```python
#!/usr/bin/env python3
import sys
import requests

if len(sys.argv) != 2:
    sys.exit(f"Usage: {sys.argv[0]} <url>")

resp = requests.get(sys.argv[1], allow_redirects=True, timeout=30)

print(f"Status: {resp.status_code}\n")
print("Headers:")
for k, v in resp.headers.items():
    print(f"  {k}: {v}")
print("\nBody (first 500 chars):")
print(resp.text[:500])
```

**How to run it:**

1. Save either version to a file, e.g. `fetch.py`.
2. Run it with a URL argument:

```bash
python3 fetch.py https://example.com
```

The stdlib version (first one) needs no dependencies. For the `requests` version, first run `pip install requests`. Both follow redirects automatically — if you want to see the final URL after redirects with `requests`, add `print(resp.url)`; with urllib, use `resp.geturl()`.
