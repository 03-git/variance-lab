```python
#!/usr/bin/env python3
"""Fetch a URL (following redirects) and print status, headers, and a body preview."""

import sys
import urllib.request
import urllib.error


def fetch(url):
    req = urllib.request.Request(url, headers={"User-Agent": "fetch-script/1.0"})
    try:
        # urllib follows redirects by default (HTTP 30x for GET)
        with urllib.request.urlopen(req) as resp:
            status = resp.status
            headers = resp.headers
            body = resp.read(500).decode("utf-8", errors="replace")
    except urllib.error.HTTPError as e:
        # HTTPError is also a response-like object
        status = e.code
        headers = e.headers
        body = e.read(500).decode("utf-8", errors="replace")

    print(f"Status code: {status}")
    print("\nResponse headers:")
    for key, value in headers.items():
        print(f"  {key}: {value}")
    print("\nBody (first 500 chars):")
    print(body)


def main():
    if len(sys.argv) != 2:
        print(f"Usage: {sys.argv[0]} <url>", file=sys.stderr)
        sys.exit(1)
    fetch(sys.argv[1])


if __name__ == "__main__":
    main()
```

**How to run it:**

Save it as `fetch.py`, then:

```bash
python3 fetch.py https://example.com
```

Notes:
- Uses only the standard library — no `pip install` needed.
- `urllib` follows redirects automatically for GET requests.
- If you'd rather use the `requests` library, the core becomes `r = requests.get(url, allow_redirects=True)` then `r.status_code`, `r.headers`, `r.text[:500]` — but you'd need `pip install requests`.
