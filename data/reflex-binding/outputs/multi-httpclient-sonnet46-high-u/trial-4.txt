```python
import sys
import urllib.request

def fetch_url(url):
    try:
        with urllib.request.urlopen(url) as response:
            print(f"Status: {response.status}")
            print("\nHeaders:")
            for key, value in response.headers.items():
                print(f"  {key}: {value}")
            body = response.read(500).decode("utf-8", errors="replace")
            print(f"\nBody (first 500 chars):\n{body}")
    except urllib.error.HTTPError as e:
        print(f"HTTP Error {e.code}: {e.reason}")
    except urllib.error.URLError as e:
        print(f"URL Error: {e.reason}")

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("Usage: python fetch.py <url>")
        sys.exit(1)
    fetch_url(sys.argv[1])
```

**How to run:**

```bash
python fetch.py https://example.com
```

Uses only the standard library (`urllib`), so no installs needed. `urlopen` follows redirects automatically. The `errors="replace"` on decode handles binary responses gracefully.
