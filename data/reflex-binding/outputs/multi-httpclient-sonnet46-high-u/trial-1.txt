```python
import sys
import urllib.request

def fetch_url(url):
    with urllib.request.urlopen(url) as response:
        status = response.status
        headers = dict(response.headers)
        body = response.read(500).decode("utf-8", errors="replace")

    print(f"Status: {status}")
    print("\nHeaders:")
    for key, value in headers.items():
        print(f"  {key}: {value}")
    print(f"\nBody (first 500 chars):\n{body}")

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("Usage: python fetch.py <url>")
        sys.exit(1)
    fetch_url(sys.argv[1])
```

`urllib.request.urlopen` follows redirects automatically. Save this as `fetch.py` and run it with:

```
python fetch.py https://example.com
```
