```python
#!/usr/bin/env python3
"""Fetch a URL and report status, headers, and a body preview."""
import sys
import urllib.request
import urllib.error


def fetch(url: str) -> None:
    req = urllib.request.Request(url, headers={"User-Agent": "url-fetch/1.0"})
    try:
        # urllib follows redirects (301/302/303/307) by default via HTTPRedirectHandler
        with urllib.request.urlopen(req) as resp:
            status = resp.status
            headers = resp.headers
            body = resp.read(500).decode(resp.headers.get_content_charset() or "utf-8", errors="replace")
            final_url = resp.geturl()
    except urllib.error.HTTPError as e:
        # HTTPError is also a response object — report what we got
        status = e.code
        headers = e.headers
        body = e.read(500).decode("utf-8", errors="replace")
        final_url = e.url
    except urllib.error.URLError as e:
        print(f"Request failed: {e.reason}", file=sys.stderr)
        sys.exit(1)

    print(f"Final URL: {final_url}")
    print(f"Status code: {status}\n")
    print("Response headers:")
    for key, value in headers.items():
        print(f"  {key}: {value}")
    print("\nFirst 500 characters of body:")
    print(body)


def main() -> None:
    if len(sys.argv) != 2:
        print(f"Usage: {sys.argv[0]} <url>", file=sys.stderr)
        sys.exit(2)
    fetch(sys.argv[1])


if __name__ == "__main__":
    main()
```

**How to run it:**

Save it as `fetch_url.py`, then:

```bash
python3 fetch_url.py https://example.com
```

Notes:
- Uses only the standard library — no `pip install` needed.
- `urllib.request.urlopen` follows redirects automatically; `resp.geturl()` shows the final URL landed on.
- HTTP error statuses (4xx/5xx) are caught and still reported rather than crashing.
- If you'd rather use the `requests` library: `r = requests.get(url, allow_redirects=True)` then print `r.status_code`, `r.headers`, and `r.text[:500]`.
