```python
#!/usr/bin/env python3
"""Fetch a URL (following redirects) and print status, headers, and body preview."""

import sys
import urllib.request
import urllib.error


def fetch(url):
    req = urllib.request.Request(url, headers={"User-Agent": "url-fetch/1.0"})
    try:
        # urllib follows redirects by default (HTTP 301/302/303/307)
        with urllib.request.urlopen(req) as resp:
            status = resp.status
            headers = resp.headers
            body = resp.read(500).decode("utf-8", errors="replace")
    except urllib.error.HTTPError as e:
        # Still useful to see the response on 4xx/5xx
        status = e.code
        headers = e.headers
        body = e.read(500).decode("utf-8", errors="replace")

    print(f"Status: {status}\n")
    print("Headers:")
    for key, value in headers.items():
        print(f"  {key}: {value}")
    print("\nBody (first 500 chars):")
    print(body)


def main():
    if len(sys.argv) != 2:
        print(f"Usage: {sys.argv[0]} <url>", file=sys.stderr)
        sys.exit(1)
    fetch(sys.argv[1])


if __name__ == "__main__":
    main()
```

How to run it:

```bash
# Save the code as fetch_url.py, then:
python3 fetch_url.py https://example.com

# Or make it executable:
chmod +x fetch_url.py
./fetch_url.py https://httpbin.org/redirect/2
```

Notes:
- Uses only the standard library — no `pip install` needed.
- `urllib.request.urlopen` follows redirects automatically; the `HTTPError` branch ensures you still see the response details for 4xx/5xx statuses.
- If you'd prefer the `requests` library (`pip install requests`), the equivalent is `r = requests.get(url)` then `r.status_code`, `r.headers`, `r.text[:500]` — `requests` follows redirects by default too (pass `allow_redirects=False` to disable).
