```python
# test_temperature.py
import pytest


def celsius_to_fahrenheit(c):
    return c * 9/5 + 32


def test_freezing_point():
    assert celsius_to_fahrenheit(0) == 32

def test_boiling_point():
    assert celsius_to_fahrenheit(100) == 212

def test_negative_forty_is_equal():
    assert celsius_to_fahrenheit(-40) == -40

def test_non_numeric_raises():
    with pytest.raises(TypeError):
        celsius_to_fahrenheit("hot")
```

Run it with:

```
pip install pytest
pytest test_temperature.py -v
```

Or without installing pytest, here's a `unittest` version that runs with just the standard library (`python test_temperature.py`):

```python
import unittest


def celsius_to_fahrenheit(c):
    return c * 9/5 + 32


class TestCelsiusToFahrenheit(unittest.TestCase):
    def test_freezing_point(self):
        self.assertEqual(celsius_to_fahrenheit(0), 32)

    def test_boiling_point(self):
        self.assertEqual(celsius_to_fahrenheit(100), 212)

    def test_negative_forty_is_equal(self):
        self.assertEqual(celsius_to_fahrenheit(-40), -40)

    def test_non_numeric_raises(self):
        with self.assertRaises(TypeError):
            celsius_to_fahrenheit("hot")


if __name__ == "__main__":
    unittest.main()
```

Note: in Python, `"hot" * 9/5 + 32` raises `TypeError` because `str / int` is unsupported (the multiplication itself succeeds, repeating the string 9 times). If you wanted to also reject numeric-looking strings like `"50"`, you'd need an explicit type check inside the function.
