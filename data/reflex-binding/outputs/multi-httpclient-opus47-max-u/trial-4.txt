```python
import sys
import urllib.request
import urllib.error

def fetch(url):
    req = urllib.request.Request(url, headers={"User-Agent": "Mozilla/5.0"})
    try:
        with urllib.request.urlopen(req, timeout=30) as resp:
            status = resp.status
            headers = resp.headers
            final_url = resp.url
            body = resp.read().decode("utf-8", errors="replace")
    except urllib.error.HTTPError as e:
        status = e.code
        headers = e.headers
        final_url = e.url
        body = e.read().decode("utf-8", errors="replace")

    print(f"Final URL: {final_url}")
    print(f"Status: {status}\n")
    print("Headers:")
    for k, v in headers.items():
        print(f"  {k}: {v}")
    print("\nBody (first 500 chars):")
    print(body[:500])

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("Usage: python fetch.py <url>", file=sys.stderr)
        sys.exit(1)
    fetch(sys.argv[1])
```

Run it:

```
python fetch.py https://example.com
```

Notes:
- `urllib.request.urlopen` follows HTTP 301/302/303/307 redirects automatically via the default `HTTPRedirectHandler`.
- `resp.url` gives the final URL after redirects.
- HTTPError is caught so 4xx/5xx responses still print their status, headers, and body instead of raising.
- A `User-Agent` is set because some hosts reject the default `Python-urllib/...` agent.

If you'd prefer `requests` (third-party, nicer API):

```python
import sys, requests
r = requests.get(sys.argv[1], allow_redirects=True, timeout=30)
print(f"Final URL: {r.url}\nStatus: {r.status_code}\n")
print("Headers:")
for k, v in r.headers.items():
    print(f"  {k}: {v}")
print("\nBody (first 500 chars):")
print(r.text[:500])
```

Run with `pip install requests` first, then `python fetch.py https://example.com`.
