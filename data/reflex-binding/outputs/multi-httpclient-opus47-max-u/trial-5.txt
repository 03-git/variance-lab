```python
import sys
import urllib.request
import urllib.error

def main():
    if len(sys.argv) != 2:
        print("Usage: python fetch.py <url>", file=sys.stderr)
        sys.exit(1)

    url = sys.argv[1]
    req = urllib.request.Request(url, headers={"User-Agent": "fetch.py/1.0"})

    try:
        with urllib.request.urlopen(req) as response:
            status = response.status
            headers = response.headers
            body = response.read(500).decode("utf-8", errors="replace")
    except urllib.error.HTTPError as e:
        status = e.code
        headers = e.headers
        body = e.read(500).decode("utf-8", errors="replace")
    except urllib.error.URLError as e:
        print(f"Request failed: {e.reason}", file=sys.stderr)
        sys.exit(1)

    print(f"Status: {status}")
    print(f"Final URL: {response.url if 'response' in dir() else url}")
    print("\nHeaders:")
    for key, value in headers.items():
        print(f"  {key}: {value}")
    print(f"\nBody (first 500 chars):\n{body}")

if __name__ == "__main__":
    main()
```

Run it with:
```bash
python fetch.py https://example.com
```

Notes:
- Uses only the standard library (no `pip install` needed).
- `urllib.request.urlopen` follows HTTP 301/302/303/307 redirects by default via the built-in `HTTPRedirectHandler`.
- Reads only the first 500 bytes of the body to avoid buffering large responses.
- Catches `HTTPError` so 4xx/5xx responses still print status, headers, and body instead of raising.

If you'd prefer the `requests` library (cleaner API, but a dependency), the equivalent is:

```python
import sys, requests
r = requests.get(sys.argv[1], allow_redirects=True, timeout=10)
print(f"Status: {r.status_code}\nFinal URL: {r.url}\n\nHeaders:")
for k, v in r.headers.items():
    print(f"  {k}: {v}")
print(f"\nBody (first 500 chars):\n{r.text[:500]}")
```
