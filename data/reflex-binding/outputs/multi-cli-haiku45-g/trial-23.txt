I'll write a Python CLI tool using `argparse` that meets your requirements:

```python
#!/usr/bin/env python3
import argparse
import json
import sys


def validate_port(value):
    """Validate that port is an integer between 1-65535."""
    try:
        port = int(value)
        if not (1 <= port <= 65535):
            raise argparse.ArgumentTypeError(f"port must be between 1-65535, got {port}")
        return port
    except ValueError:
        raise argparse.ArgumentTypeError(f"port must be an integer, got '{value}'")


def main():
    parser = argparse.ArgumentParser(
        description="Display parsed configuration as JSON"
    )
    parser.add_argument(
        "--host",
        default="localhost",
        help="Host to connect to (default: localhost)"
    )
    parser.add_argument(
        "--port",
        type=validate_port,
        default=8080,
        help="Port number 1-65535 (default: 8080)"
    )
    parser.add_argument(
        "--verbose",
        action="store_true",
        help="Enable verbose output"
    )

    args = parser.parse_args()

    config = {
        "host": args.host,
        "port": args.port,
        "verbose": args.verbose
    }

    print(json.dumps(config, indent=2))


if __name__ == "__main__":
    main()
```

**How to run it:**

```bash
# Basic usage
python3 cli.py

# With custom host and port
python3 cli.py --host example.com --port 3000

# With verbose flag
python3 cli.py --host 0.0.0.0 --port 5000 --verbose

# Test validation (invalid port)
python3 cli.py --port 99999
# Output: usage: cli.py [-h] [--host HOST] [--port PORT] [--verbose]
#         cli.py: error: argument --port: port must be between 1-65535, got 99999

# Test validation (non-integer port)
python3 cli.py --port abc
# Output: usage: cli.py [-h] [--host HOST] [--port PORT] [--verbose]
#         cli.py: error: argument --port: port must be an integer, got 'abc'
```
