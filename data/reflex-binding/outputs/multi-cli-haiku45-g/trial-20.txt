Here's a Python CLI tool that meets your requirements:

```python
import argparse
import json
import sys


def validate_port(value):
    """Validate that port is an integer between 1-65535."""
    try:
        port = int(value)
    except ValueError:
        raise argparse.ArgumentTypeError(f"Port must be an integer, got: {value}")
    
    if not (1 <= port <= 65535):
        raise argparse.ArgumentTypeError(f"Port must be between 1-65535, got: {port}")
    
    return port


def main():
    parser = argparse.ArgumentParser(
        description="Parse and display configuration as JSON"
    )
    parser.add_argument(
        "--host",
        default="localhost",
        help="Host address (default: localhost)"
    )
    parser.add_argument(
        "--port",
        type=validate_port,
        default=8080,
        help="Port number, must be 1-65535 (default: 8080)"
    )
    parser.add_argument(
        "--verbose",
        action="store_true",
        help="Enable verbose output"
    )
    
    args = parser.parse_args()
    
    config = {
        "host": args.host,
        "port": args.port,
        "verbose": args.verbose
    }
    
    print(json.dumps(config, indent=2))


if __name__ == "__main__":
    main()
```

**How to run it:**

```bash
# Basic usage (uses defaults)
python cli.py
# Output: {"host": "localhost", "port": 8080, "verbose": false}

# With custom host and port
python cli.py --host 0.0.0.0 --port 3000
# Output: {"host": "0.0.0.0", "port": 3000, "verbose": false}

# With verbose flag
python cli.py --host example.com --port 443 --verbose
# Output: {"host": "example.com", "port": 443, "verbose": true}

# Test validation (invalid port)
python cli.py --port 70000
# Error: usage: cli.py [-h] [--host HOST] [--port PORT] [--verbose]
#        cli.py: error: argument --port: Port must be between 1-65535, got: 70000

# Test validation (non-integer port)
python cli.py --port abc
# Error: argument --port: Port must be an integer, got: abc
```

The script uses `argparse` for clean command-line parsing, a custom `validate_port()` function to enforce the integer constraint and range check, and outputs the configuration as formatted JSON.
