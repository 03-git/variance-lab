Here's a Python CLI tool using `argparse` for flag parsing and `json` for output:

```python
import argparse
import json
import sys


def validate_port(port_str):
    """Validate that port is an integer between 1-65535."""
    try:
        port = int(port_str)
        if port < 1 or port > 65535:
            raise argparse.ArgumentTypeError(f"port must be between 1-65535, got {port}")
        return port
    except ValueError:
        raise argparse.ArgumentTypeError(f"port must be an integer, got '{port_str}'")


def main():
    parser = argparse.ArgumentParser(
        description="Parse and display configuration as JSON"
    )
    parser.add_argument(
        "--host",
        default="localhost",
        help="Host address (default: localhost)"
    )
    parser.add_argument(
        "--port",
        type=validate_port,
        default=8080,
        help="Port number 1-65535 (default: 8080)"
    )
    parser.add_argument(
        "--verbose",
        action="store_true",
        help="Enable verbose output"
    )

    args = parser.parse_args()

    config = {
        "host": args.host,
        "port": args.port,
        "verbose": args.verbose
    }

    print(json.dumps(config, indent=2))


if __name__ == "__main__":
    main()
```

**How to run it:**

```bash
# Basic usage (uses defaults)
python cli_tool.py
# Output:
# {
#   "host": "localhost",
#   "port": 8080,
#   "verbose": false
# }

# With custom host and port
python cli_tool.py --host example.com --port 3000
# Output:
# {
#   "host": "example.com",
#   "port": 3000,
#   "verbose": false
# }

# With verbose flag
python cli_tool.py --host 0.0.0.0 --port 5000 --verbose
# Output:
# {
#   "host": "0.0.0.0",
#   "port": 5000,
#   "verbose": true
# }

# Invalid port (shows error)
python cli_tool.py --port 99999
# Error: argument --port: port must be between 1-65535, got 99999

# Non-integer port (shows error)
python cli_tool.py --port abc
# Error: argument --port: port must be an integer, got 'abc'
```

The `validate_port()` function acts as a custom type converter for argparse, running validation immediately during argument parsing so invalid ports are caught before JSON output.
