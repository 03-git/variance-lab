Here's a Python CLI tool that meets your requirements:

```python
#!/usr/bin/env python3
import argparse
import json
import sys

def validate_port(value):
    """Validate port is an integer between 1-65535."""
    try:
        port = int(value)
        if not (1 <= port <= 65535):
            raise argparse.ArgumentTypeError(
                f"port must be between 1 and 65535, got {port}"
            )
        return port
    except ValueError:
        raise argparse.ArgumentTypeError(f"port must be an integer, got '{value}'")

def main():
    parser = argparse.ArgumentParser(
        description="Parse and display host/port configuration as JSON"
    )
    parser.add_argument(
        "--host",
        default="localhost",
        help="hostname or IP address (default: localhost)"
    )
    parser.add_argument(
        "--port",
        type=validate_port,
        default=8080,
        help="port number 1-65535 (default: 8080)"
    )
    parser.add_argument(
        "--verbose",
        action="store_true",
        help="enable verbose output"
    )
    
    args = parser.parse_args()
    
    config = {
        "host": args.host,
        "port": args.port,
        "verbose": args.verbose
    }
    
    print(json.dumps(config, indent=2))

if __name__ == "__main__":
    main()
```

**How to run it:**

```bash
# Basic usage with defaults
python3 script.py
# Output: {"host": "localhost", "port": 8080, "verbose": false}

# Custom host and port
python3 script.py --host 192.168.1.1 --port 3000
# Output: {"host": "192.168.1.1", "port": 3000, "verbose": false}

# With verbose flag
python3 script.py --host example.com --port 443 --verbose
# Output: {"host": "example.com", "port": 443, "verbose": true}

# Invalid port (shows error)
python3 script.py --port 70000
# Error: usage: script.py [-h] [--host HOST] [--port PORT] [--verbose]
# script.py: error: argument --port: port must be between 1 and 65535, got 70000

# Invalid port type (shows error)
python3 script.py --port abc
# Error: usage: script.py [-h] [--host HOST] [--port PORT] [--verbose]
# script.py: error: argument --port: port must be an integer, got 'abc'

# Help
python3 script.py --help
```

The `validate_port` function ensures the port is an integer and within the valid range (1-65535). If validation fails, argparse automatically displays a user-friendly error message and exits.
