```python
# test_temperature.py
import unittest

def celsius_to_fahrenheit(c):
    return c * 9/5 + 32

class TestCelsiusToFahrenheit(unittest.TestCase):

    def test_freezing_point(self):
        self.assertEqual(celsius_to_fahrenheit(0), 32)

    def test_boiling_point(self):
        self.assertEqual(celsius_to_fahrenheit(100), 212)

    def test_equal_point(self):
        # -40 is the point where Celsius and Fahrenheit are equal
        self.assertEqual(celsius_to_fahrenheit(-40), -40)

    def test_non_numeric_input(self):
        with self.assertRaises(TypeError):
            celsius_to_fahrenheit("hot")

if __name__ == "__main__":
    unittest.main()
```

**To run it**, save the code to `test_temperature.py`, then:

```bash
# Run all tests
python -m unittest test_temperature

# Run with verbose output
python -m unittest test_temperature -v

# Or directly
python test_temperature.py
```

The `TypeError` test works because Python raises one naturally when multiplying a string by a float — no special handling needed in the function itself.
