```python
#!/usr/bin/env python3
"""Fetch a URL, follow redirects, and print status, headers, and body preview."""
import sys
import urllib.request
import urllib.error


def main():
    if len(sys.argv) != 2:
        print(f"Usage: {sys.argv[0]} <url>", file=sys.stderr)
        sys.exit(1)

    url = sys.argv[1]

    # urllib follows redirects by default (HTTPRedirectHandler).
    req = urllib.request.Request(url, headers={"User-Agent": "fetch-script/1.0"})
    try:
        with urllib.request.urlopen(req) as resp:
            status = resp.status
            headers = resp.headers
            body = resp.read()
            final_url = resp.geturl()
    except urllib.error.HTTPError as e:
        # HTTPError is also a response-like object.
        status = e.code
        headers = e.headers
        body = e.read()
        final_url = e.url
    except urllib.error.URLError as e:
        print(f"Request failed: {e.reason}", file=sys.stderr)
        sys.exit(1)

    print(f"Final URL: {final_url}")
    print(f"Status code: {status}\n")

    print("Response headers:")
    for name, value in headers.items():
        print(f"  {name}: {value}")

    text = body.decode("utf-8", errors="replace")
    print(f"\nBody (first 500 chars):\n{text[:500]}")


if __name__ == "__main__":
    main()
```

Run it:

```bash
# Save as fetch.py, then:
python3 fetch.py https://example.com

# A URL that redirects (ends up at the final destination):
python3 fetch.py http://github.com
```

Notes:
- Uses only the standard library — no `pip install` needed.
- `urllib` follows redirects automatically; `resp.geturl()` shows where you landed.
- If you'd rather use `requests`: `r = requests.get(url, allow_redirects=True)` then `r.status_code`, `r.headers`, `r.text[:500]` — but you'd need `pip install requests`.
