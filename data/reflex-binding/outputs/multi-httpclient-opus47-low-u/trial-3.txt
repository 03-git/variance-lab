```python
#!/usr/bin/env python3
"""Fetch a URL and print status, headers, and a body preview."""
import sys
import urllib.request
import urllib.error


def main():
    if len(sys.argv) != 2:
        print(f"Usage: {sys.argv[0]} <url>", file=sys.stderr)
        sys.exit(1)

    url = sys.argv[1]
    req = urllib.request.Request(url, headers={"User-Agent": "fetch-script/1.0"})

    try:
        # urllib follows HTTP redirects by default
        with urllib.request.urlopen(req) as resp:
            status = resp.status
            headers = resp.headers
            body = resp.read(500).decode("utf-8", errors="replace")
    except urllib.error.HTTPError as e:
        # HTTPError is also a response-like object
        status = e.code
        headers = e.headers
        body = e.read(500).decode("utf-8", errors="replace")
    except urllib.error.URLError as e:
        print(f"Request failed: {e.reason}", file=sys.stderr)
        sys.exit(1)

    print(f"Status: {status}")
    print(f"Final URL: {resp.url if 'resp' in dir() else url}")
    print("\nHeaders:")
    for key, value in headers.items():
        print(f"  {key}: {value}")
    print("\nBody (first 500 chars):")
    print(body)


if __name__ == "__main__":
    main()
```

If you prefer the `requests` library (cleaner, but a third-party dependency):

```python
#!/usr/bin/env python3
import sys
import requests

if len(sys.argv) != 2:
    sys.exit(f"Usage: {sys.argv[0]} <url>")

resp = requests.get(sys.argv[1], allow_redirects=True, timeout=30)

print(f"Status: {resp.status_code}")
print(f"Final URL: {resp.url}")
print("\nHeaders:")
for key, value in resp.headers.items():
    print(f"  {key}: {value}")
print("\nBody (first 500 chars):")
print(resp.text[:500])
```

## How to run

Save either version to a file, e.g. `fetch.py`, then:

```bash
python3 fetch.py https://example.com
```

- The stdlib version needs no installation.
- For the `requests` version: `pip install requests` first.
- `urllib`/`requests` both follow redirects automatically; the script prints the **final** URL after any redirect chain.

Example:

```bash
python3 fetch.py https://httpbin.org/redirect/2
```

This will follow the two redirects and show the final `200` response.
