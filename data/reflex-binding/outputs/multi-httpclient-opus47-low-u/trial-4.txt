```python
#!/usr/bin/env python3
"""Fetch a URL, follow redirects, and print status, headers, and a body preview."""

import sys
import urllib.request
import urllib.error


def fetch(url):
    req = urllib.request.Request(url, headers={"User-Agent": "url-fetch/1.0"})
    try:
        # urllib follows redirects (3xx) automatically via HTTPRedirectHandler
        with urllib.request.urlopen(req) as resp:
            status = resp.status
            headers = resp.headers
            body = resp.read(2000).decode(resp.headers.get_content_charset() or "utf-8", errors="replace")
            final_url = resp.geturl()
    except urllib.error.HTTPError as e:
        # HTTPError is also a response-like object
        status = e.code
        headers = e.headers
        body = e.read(2000).decode("utf-8", errors="replace")
        final_url = e.url
    except urllib.error.URLError as e:
        print(f"Request failed: {e.reason}", file=sys.stderr)
        sys.exit(1)

    print(f"Final URL: {final_url}")
    print(f"Status code: {status}\n")
    print("Response headers:")
    for key, value in headers.items():
        print(f"  {key}: {value}")
    print("\nBody (first 500 chars):")
    print(body[:500])


def main():
    if len(sys.argv) != 2:
        print(f"Usage: {sys.argv[0]} <url>", file=sys.stderr)
        sys.exit(2)
    fetch(sys.argv[1])


if __name__ == "__main__":
    main()
```

If you'd rather use `requests` (cleaner, but a third-party dependency):

```python
#!/usr/bin/env python3
import sys
import requests

if len(sys.argv) != 2:
    sys.exit(f"Usage: {sys.argv[0]} <url>")

resp = requests.get(sys.argv[1], allow_redirects=True, timeout=30,
                    headers={"User-Agent": "url-fetch/1.0"})

print(f"Final URL: {resp.url}")
print(f"Status code: {resp.status_code}\n")
print("Response headers:")
for k, v in resp.headers.items():
    print(f"  {k}: {v}")
print("\nBody (first 500 chars):")
print(resp.text[:500])
```

## How to run it

1. Save either version to a file, e.g. `fetch_url.py`.
2. The stdlib version needs nothing extra. For the `requests` version: `pip install requests`.
3. Run it:

```bash
python3 fetch_url.py https://example.com
```

Notes:
- `urllib.request.urlopen` follows HTTP redirects automatically, so the printed status is the *final* response's status (e.g. `200`), not the intermediate `301`/`302`. `resp.geturl()` shows where you ended up.
- 4xx/5xx responses raise `HTTPError` in the stdlib version; the script catches it so you still see the status, headers, and body.
- Add `python3 fetch_url.py` with no arguments to see the usage message.
