```python
#!/usr/bin/env python3
import sys
import urllib.request

def main():
    if len(sys.argv) != 2:
        print(f"Usage: {sys.argv[0]} <URL>", file=sys.stderr)
        sys.exit(1)

    url = sys.argv[1]

    # urllib follows HTTP redirects by default via HTTPRedirectHandler
    req = urllib.request.Request(url, headers={"User-Agent": "redirect-fetcher/1.0"})
    try:
        with urllib.request.urlopen(req) as resp:
            body = resp.read(500).decode("utf-8", errors="replace")
            print(f"Status: {resp.status} {resp.reason}")
            print(f"Final URL: {resp.url}")
            print("\nHeaders:")
            for key, value in resp.headers.items():
                print(f"  {key}: {value}")
            print("\nBody (first 500 chars):")
            print(body)
    except urllib.error.HTTPError as e:
        # HTTPError is also a response object
        print(f"Status: {e.code} {e.reason}")
        print("\nHeaders:")
        for key, value in e.headers.items():
            print(f"  {key}: {value}")
        print("\nBody (first 500 chars):")
        print(e.read(500).decode("utf-8", errors="replace"))
    except urllib.error.URLError as e:
        print(f"Request failed: {e.reason}", file=sys.stderr)
        sys.exit(1)

if __name__ == "__main__":
    main()
```

If you'd rather use `requests`:

```python
#!/usr/bin/env python3
import sys
import requests

def main():
    if len(sys.argv) != 2:
        print(f"Usage: {sys.argv[0]} <URL>", file=sys.stderr)
        sys.exit(1)

    resp = requests.get(sys.argv[1], allow_redirects=True, timeout=30)
    print(f"Status: {resp.status_code}")
    print(f"Final URL: {resp.url}")
    print("\nHeaders:")
    for key, value in resp.headers.items():
        print(f"  {key}: {value}")
    print("\nBody (first 500 chars):")
    print(resp.text[:500])

if __name__ == "__main__":
    main()
```

**How to run it:**

1. Save either version to a file, e.g. `fetch.py`.
2. (Only for the `requests` version) install the dependency: `pip install requests`
3. Run it with a URL argument:

```bash
python3 fetch.py https://example.com
```

The stdlib version (first one) needs no installation. `urllib.request.urlopen` follows redirects automatically, and `resp.url` shows where you ended up after any redirect hops.
