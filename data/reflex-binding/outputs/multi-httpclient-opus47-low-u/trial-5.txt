```python
#!/usr/bin/env python3
"""Fetch a URL (following redirects) and print status, headers, and a body preview."""

import sys
import urllib.request
import urllib.error


def main():
    if len(sys.argv) != 2:
        print(f"Usage: {sys.argv[0]} <url>", file=sys.stderr)
        sys.exit(1)

    url = sys.argv[1]

    # urllib follows redirects by default for HTTP GET via HTTPRedirectHandler.
    req = urllib.request.Request(url, headers={"User-Agent": "fetch-script/1.0"})

    try:
        with urllib.request.urlopen(req) as resp:
            status = resp.status
            headers = resp.headers
            body = resp.read(500).decode("utf-8", errors="replace")
    except urllib.error.HTTPError as e:
        # HTTPError is also a response-like object.
        status = e.code
        headers = e.headers
        body = e.read(500).decode("utf-8", errors="replace")
    except urllib.error.URLError as e:
        print(f"Request failed: {e.reason}", file=sys.stderr)
        sys.exit(1)

    print(f"Status code: {status}\n")
    print("Response headers:")
    for key, value in headers.items():
        print(f"  {key}: {value}")
    print("\nBody (first 500 chars):")
    print(body)


if __name__ == "__main__":
    main()
```

How to run it:

```bash
# Save the code above as fetch.py, then:
python3 fetch.py https://example.com

# or make it executable:
chmod +x fetch.py
./fetch.py https://httpbin.org/redirect/2
```

Notes:
- Uses only the standard library — no `pip install` needed.
- Redirects are followed automatically by `urllib`. The `headers` printed are from the *final* response in the chain.
- If you'd prefer the `requests` library (`pip install requests`), the equivalent core is `r = requests.get(url, timeout=10)` then `r.status_code`, `r.headers`, `r.text[:500]` — `requests` follows redirects by default too (`allow_redirects=True`).
