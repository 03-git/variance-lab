Here's a script using only the standard library — no pip install needed.

```python
#!/usr/bin/env python3
import urllib.request
from html.parser import HTMLParser

class LinkExtractor(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []
        self.current_href = None
        self.current_text = []

    def handle_starttag(self, tag, attrs):
        if tag == "a":
            href = dict(attrs).get("href")
            if href is not None:
                self.current_href = href
                self.current_text = []

    def handle_data(self, data):
        if self.current_href is not None:
            self.current_text.append(data)

    def handle_endtag(self, tag):
        if tag == "a" and self.current_href is not None:
            text = "".join(self.current_text).strip()
            self.links.append((self.current_href, text))
            self.current_href = None
            self.current_text = []

req = urllib.request.Request(
    "https://example.com",
    headers={"User-Agent": "Mozilla/5.0 (link-extractor)"},
)
with urllib.request.urlopen(req, timeout=10) as resp:
    html = resp.read().decode(resp.headers.get_content_charset() or "utf-8")

parser = LinkExtractor()
parser.feed(html)

for href, text in parser.links:
    print(f"{href}\t{text}")
```

Run it:

```
python3 extract_links.py
```

Against `https://example.com` you'll get one line: `https://www.iana.org/domains/example` and the text `More information...`. Output is tab-separated so you can pipe to `cut -f1` for hrefs only.
