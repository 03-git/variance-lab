Here's a Python script that does what you need:

```python
import sys
import requests

def fetch_url(url):
    try:
        response = requests.get(url, timeout=10)
        
        print(f"Status Code: {response.status_code}\n")
        
        print("Response Headers:")
        for header, value in response.headers.items():
            print(f"  {header}: {value}")
        
        print(f"\nFirst 500 characters of body:")
        body = response.text[:500]
        print(body)
        
        if len(response.text) > 500:
            print(f"\n... (truncated, total length: {len(response.text)} characters)")
    
    except requests.exceptions.MissingSchema:
        print("Error: Invalid URL. Please include http:// or https://")
    except requests.exceptions.ConnectionError:
        print(f"Error: Could not connect to {url}")
    except requests.exceptions.Timeout:
        print("Error: Request timed out")
    except Exception as e:
        print(f"Error: {e}")

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("Usage: python script.py <URL>")
        sys.exit(1)
    
    url = sys.argv[1]
    fetch_url(url)
```

**How to run it:**

```bash
python script.py https://example.com
```
