I'll show you a Python script that fetches a URL and extracts all hyperlinks.

```python
import requests
from html.parser import HTMLParser
from urllib.parse import urljoin

class LinkExtractor(HTMLParser):
    def __init__(self, base_url):
        super().__init__()
        self.base_url = base_url
        self.links = []
    
    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            href = None
            for attr, value in attrs:
                if attr == 'href':
                    href = value
                    break
            if href:
                # Convert relative URLs to absolute
                absolute_url = urljoin(self.base_url, href)
                self.links.append(absolute_url)
    
    def handle_data(self, data):
        if self.link_text_buffer is not None:
            self.link_text_buffer += data.strip()
    
    def get_links(self):
        return self.links

class LinkExtractorWithText(HTMLParser):
    def __init__(self, base_url):
        super().__init__()
        self.base_url = base_url
        self.links = []
        self.current_link = None
        self.link_text = []
    
    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            for attr, value in attrs:
                if attr == 'href':
                    self.current_link = urljoin(self.base_url, value)
                    self.link_text = []
                    break
    
    def handle_data(self, data):
        if self.current_link is not None:
            self.link_text.append(data.strip())
    
    def handle_endtag(self, tag):
        if tag == 'a' and self.current_link is not None:
            text = ' '.join(self.link_text).strip()
            self.links.append({
                'href': self.current_link,
                'text': text if text else '(no text)'
            })
            self.current_link = None
            self.link_text = []

def fetch_and_extract_links(url):
    try:
        response = requests.get(url, timeout=10)
        response.raise_for_status()
        
        parser = LinkExtractorWithText(url)
        parser.feed(response.text)
        
        return parser.links
    except requests.RequestException as e:
        print(f"Error fetching {url}: {e}")
        return []

if __name__ == '__main__':
    url = 'https://example.com'
    links = fetch_and_extract_links(url)
    
    if links:
        print(f"Found {len(links)} links:\n")
        for i, link in enumerate(links, 1):
            print(f"{i}. {link['text']}")
            print(f"   URL: {link['href']}\n")
    else:
        print("No links found or error occurred.")
```

**How to run it:**

1. Install the required library:
```bash
pip install requests
```

2. Run the script:
```bash
python script.py
```

The script will:
- Fetch the HTML from `https://example.com`
- Parse all `<a>` tags
- Extract both the href and the link text
- Convert relative URLs to absolute URLs
- Print each link with its text

You can modify the `url` variable to fetch from any website. The output shows the link text first, then the full URL on the next line.
