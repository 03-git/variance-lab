```python
import unittest

def celsius_to_fahrenheit(c):
    return c * 9/5 + 32

class TestCelsiusToFahrenheit(unittest.TestCase):

    def test_freezing(self):
        self.assertEqual(celsius_to_fahrenheit(0), 32)

    def test_boiling(self):
        self.assertEqual(celsius_to_fahrenheit(100), 212)

    def test_negative_forty(self):
        self.assertEqual(celsius_to_fahrenheit(-40), -40)

    def test_non_numeric(self):
        with self.assertRaises(TypeError):
            celsius_to_fahrenheit("hot")

if __name__ == "__main__":
    unittest.main()
```

Save as `test_temp.py` and run:

```
python -m unittest test_temp.py -v
```

Or run inline without a file:

```
python -c "
import unittest

def celsius_to_fahrenheit(c):
    return c * 9/5 + 32

class T(unittest.TestCase):
    def test_freezing(self):   self.assertEqual(celsius_to_fahrenheit(0), 32)
    def test_boiling(self):    self.assertEqual(celsius_to_fahrenheit(100), 212)
    def test_neg40(self):      self.assertEqual(celsius_to_fahrenheit(-40), -40)
    def test_non_numeric(self):
        with self.assertRaises(TypeError): celsius_to_fahrenheit('hot')

unittest.main(argv=[''], verbosity=2, exit=False)
"
```

Note on the non-numeric test: `celsius_to_fahrenheit("hot")` raises `TypeError` naturally because Python can't multiply a string by a float — no explicit guard needed in the function.
