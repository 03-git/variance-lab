```python
import http.server, json, datetime

class H(http.server.BaseHTTPRequestHandler):
    def do_GET(self):
        if self.path != "/time":
            self.send_response(404); self.end_headers(); return
        body = json.dumps({"time": datetime.datetime.now(datetime.timezone.utc).isoformat()}).encode()
        self.send_response(200)
        self.send_header("Content-Type", "application/json")
        self.send_header("Content-Length", str(len(body)))
        self.end_headers()
        self.wfile.write(body)

http.server.HTTPServer(("", 8000), H).serve_forever()
```

Run: `python3 server.py` → `curl localhost:8000/time`
