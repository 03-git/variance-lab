```python
import sys
import urllib.request

def main():
    if len(sys.argv) != 2:
        print("Usage: python fetch.py <url>", file=sys.stderr)
        sys.exit(1)

    url = sys.argv[1]
    req = urllib.request.Request(url, headers={"User-Agent": "fetch.py/1.0"})

    with urllib.request.urlopen(req) as resp:
        body = resp.read().decode("utf-8", errors="replace")

        print(f"Status: {resp.status} {resp.reason}")
        print(f"Final URL: {resp.url}")
        print()
        print("Headers:")
        for key, value in resp.headers.items():
            print(f"  {key}: {value}")
        print()
        print("Body (first 500 chars):")
        print(body[:500])

if __name__ == "__main__":
    main()
```

`urllib.request.urlopen` follows HTTP 30x redirects by default via `HTTPRedirectHandler`, so no extra config is needed.

Run it:

```bash
python fetch.py https://example.com
```
