```python
import urllib.request
from html.parser import HTMLParser

class LinkParser(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []
        self._current_href = None
        self._current_text = []

    def handle_starttag(self, tag, attrs):
        if tag == "a":
            attrs_dict = dict(attrs)
            self._current_href = attrs_dict.get("href")
            self._current_text = []

    def handle_data(self, data):
        if self._current_href is not None:
            self._current_text.append(data)

    def handle_endtag(self, tag):
        if tag == "a" and self._current_href is not None:
            self.links.append((self._current_href, "".join(self._current_text).strip()))
            self._current_href = None
            self._current_text = []

with urllib.request.urlopen("https://example.com") as resp:
    html = resp.read().decode("utf-8")

parser = LinkParser()
parser.feed(html)

for href, text in parser.links:
    print(f"{text!r:30} -> {href}")
```

**To run** (no dependencies beyond the stdlib):

```bash
python3 script.py
```

Uses only `urllib` and `html.parser` from the standard library — no `requests` or `beautifulsoup4` needed. `example.com` has one link (`More information...` -> `https://www.iana.org/domains/reserved`), so that's what you'll see.
