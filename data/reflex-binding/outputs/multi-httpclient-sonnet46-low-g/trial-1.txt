```python
import sys
import urllib.request

def fetch(url):
    with urllib.request.urlopen(url) as r:
        print(f"Status: {r.status}")
        print("\nHeaders:")
        for k, v in r.headers.items():
            print(f"  {k}: {v}")
        print("\nBody (first 500 chars):")
        print(r.read(500).decode("utf-8", errors="replace"))

fetch(sys.argv[1])
```

Run it:

```bash
python3 script.py https://example.com
```

`urllib.request.urlopen` follows redirects by default. No third-party dependencies needed.
