```python
import urllib.request
from html.parser import HTMLParser

class LinkExtractor(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []
        self._href = None
        self._text = []

    def handle_starttag(self, tag, attrs):
        if tag == "a":
            self._href = dict(attrs).get("href")
            self._text = []

    def handle_data(self, data):
        if self._href is not None:
            self._text.append(data)

    def handle_endtag(self, tag):
        if tag == "a" and self._href is not None:
            self.links.append((self._href, "".join(self._text).strip()))
            self._href = None
            self._text = []

def main():
    url = "https://example.com"
    req = urllib.request.Request(url, headers={"User-Agent": "Mozilla/5.0"})
    with urllib.request.urlopen(req) as resp:
        html = resp.read().decode(resp.headers.get_content_charset() or "utf-8")

    parser = LinkExtractor()
    parser.feed(html)

    for href, text in parser.links:
        print(f"{href}\t{text}")

if __name__ == "__main__":
    main()
```

Run it:

```
python3 script.py
```

Uses only the stdlib — no `pip install` needed. For `https://example.com` you should see one link: `https://www.iana.org/domains/example` with text `More information...`.
