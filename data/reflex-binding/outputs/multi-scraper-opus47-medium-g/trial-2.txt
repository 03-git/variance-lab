Kernel primitive: this needs no third-party library — Python stdlib `urllib.request` (HTTP over sockets) + `html.parser.HTMLParser` (the parsing primitive) cover it. No `requests`, no `beautifulsoup4`.

```python
#!/usr/bin/env python3
"""Fetch a URL and print every <a> href with its link text. Stdlib only."""
import sys
from urllib.request import urlopen, Request
from html.parser import HTMLParser


class LinkExtractor(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []          # list of [href, text]
        self._href = None        # href of the <a> we're inside, or None

    def handle_starttag(self, tag, attrs):
        if tag == "a":
            self._href = dict(attrs).get("href")
            if self._href is not None:
                self.links.append([self._href, ""])

    def handle_data(self, data):
        if self._href is not None and self.links:
            self.links[-1][1] += data.strip()

    def handle_endtag(self, tag):
        if tag == "a":
            self._href = None


def main(url):
    req = Request(url, headers={"User-Agent": "link-extractor/1.0"})
    with urlopen(req, timeout=10) as resp:
        charset = resp.headers.get_content_charset() or "utf-8"
        html = resp.read().decode(charset, errors="replace")

    parser = LinkExtractor()
    parser.feed(html)

    for href, text in parser.links:
        print(f"{href}\t{text or '(no text)'}")


if __name__ == "__main__":
    main(sys.argv[1] if len(sys.argv) > 1 else "https://example.com")
```

Run it:

```bash
python3 link_extractor.py                 # defaults to https://example.com
python3 link_extractor.py https://news.ycombinator.com   # or pass any URL
```

Output for `https://example.com` is a single link:

```
https://www.iana.org/domains/example	More information...
```

Notes:
- Relative hrefs print as-is; wrap with `urllib.parse.urljoin(url, href)` if you want them absolutized.
- `HTMLParser` is lenient with messy markup but won't execute JS — links injected client-side won't appear.
