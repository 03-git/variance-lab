```python
import sys
import urllib.request
import urllib.error

def main():
    if len(sys.argv) != 2:
        print("Usage: python fetch.py <URL>", file=sys.stderr)
        sys.exit(1)

    url = sys.argv[1]
    req = urllib.request.Request(url, headers={"User-Agent": "fetch.py/1.0"})

    try:
        with urllib.request.urlopen(req, timeout=30) as resp:
            status = resp.status
            headers = resp.headers
            body = resp.read(500).decode(resp.headers.get_content_charset() or "utf-8", errors="replace")
    except urllib.error.HTTPError as e:
        status = e.code
        headers = e.headers
        body = e.read(500).decode("utf-8", errors="replace")

    print(f"Status: {status}")
    print("Headers:")
    for k, v in headers.items():
        print(f"  {k}: {v}")
    print("Body (first 500 chars):")
    print(body)

if __name__ == "__main__":
    main()
```

Run it:

```
python fetch.py https://example.com
```

Notes:
- `urllib.request.urlopen` follows HTTP 30x redirects by default via `HTTPRedirectHandler`.
- `HTTPError` is caught so 4xx/5xx still print status, headers, and body instead of crashing.
- `resp.read(500)` reads at most 500 bytes; for multi-byte encodings the decoded string may be slightly shorter, which is fine for a preview.
