```python
#!/usr/bin/env python3
import sys
from urllib.request import Request, urlopen
from urllib.error import HTTPError, URLError

def main():
    if len(sys.argv) != 2:
        print("usage: fetch.py <url>", file=sys.stderr)
        sys.exit(2)

    url = sys.argv[1]
    req = Request(url, headers={"User-Agent": "fetch.py/1.0"})

    try:
        with urlopen(req, timeout=30) as resp:
            body = resp.read()
            status = resp.status
            headers = resp.headers
            final_url = resp.url
    except HTTPError as e:
        body = e.read()
        status = e.code
        headers = e.headers
        final_url = e.url
    except URLError as e:
        print(f"request failed: {e.reason}", file=sys.stderr)
        sys.exit(1)

    print(f"Final URL: {final_url}")
    print(f"Status: {status}")
    print("Headers:")
    for k, v in headers.items():
        print(f"  {k}: {v}")

    charset = headers.get_content_charset() or "utf-8"
    text = body.decode(charset, errors="replace")
    print("\nBody (first 500 chars):")
    print(text[:500])

if __name__ == "__main__":
    main()
```

Run it:

```bash
python3 fetch.py https://example.com
```

Notes:
- `urllib.request.urlopen` follows HTTP 30x redirects by default via the built-in `HTTPRedirectHandler`.
- Stdlib only — no `pip install` needed.
- `HTTPError` is caught so 4xx/5xx responses still print status, headers, and body instead of raising.
- `resp.url` shows the final URL after any redirects.
