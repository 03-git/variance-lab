```python
#!/usr/bin/env python3
import sys
import urllib.request
import urllib.error

def main():
    if len(sys.argv) != 2:
        print("Usage: fetch.py <url>", file=sys.stderr)
        sys.exit(2)

    url = sys.argv[1]
    req = urllib.request.Request(url, headers={"User-Agent": "fetch.py/1.0"})

    try:
        with urllib.request.urlopen(req) as resp:
            status = resp.status
            headers = resp.headers
            body = resp.read(500).decode(resp.headers.get_content_charset() or "utf-8", errors="replace")
            final_url = resp.geturl()
    except urllib.error.HTTPError as e:
        status = e.code
        headers = e.headers
        body = e.read(500).decode("utf-8", errors="replace")
        final_url = e.url

    print(f"Final URL: {final_url}")
    print(f"Status: {status}")
    print("Headers:")
    for k, v in headers.items():
        print(f"  {k}: {v}")
    print("Body (first 500 chars):")
    print(body)

if __name__ == "__main__":
    main()
```

Run it:

```bash
python3 fetch.py https://example.com
```

`urllib.request.urlopen` follows 301/302/303/307 redirects automatically. `HTTPError` is caught so 4xx/5xx responses still print headers and body instead of raising.
