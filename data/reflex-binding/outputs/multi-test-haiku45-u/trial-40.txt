I'll show you a comprehensive Python test suite for the `celsius_to_fahrenheit` function using pytest, along with how to run it.

```python
# test_celsius_to_fahrenheit.py

import pytest


def celsius_to_fahrenheit(c):
    """Convert Celsius to Fahrenheit."""
    return c * 9/5 + 32


class TestCelsiusToFahrenheit:
    """Test suite for celsius_to_fahrenheit function."""
    
    def test_freezing_point(self):
        """0°C should equal 32°F (freezing point of water)."""
        assert celsius_to_fahrenheit(0) == 32
    
    def test_boiling_point(self):
        """100°C should equal 212°F (boiling point of water)."""
        assert celsius_to_fahrenheit(100) == 212
    
    def test_same_in_both_scales(self):
        """-40°C should equal -40°F (same in both scales)."""
        assert celsius_to_fahrenheit(-40) == -40
    
    def test_non_numeric_input(self):
        """Non-numeric input should raise TypeError."""
        with pytest.raises(TypeError):
            celsius_to_fahrenheit("32")
    
    def test_float_input(self):
        """Float input should work correctly."""
        assert celsius_to_fahrenheit(37) == pytest.approx(98.6)
```

**How to run it:**

```bash

# Run the tests

# Or with verbose output showing test names
pytest test_celsius_to_fahrenheit.py -v

# Or run a specific test
```
