I'll show you a comprehensive Python test suite for the `celsius_to_fahrenheit` function using pytest:

```python
import pytest

def celsius_to_fahrenheit(c):
    return c * 9/5 + 32

# Test cases
def test_freezing_point():
    """Test conversion at freezing point of water."""
    assert celsius_to_fahrenheit(0) == 32

def test_boiling_point():
    """Test conversion at boiling point of water."""
    assert celsius_to_fahrenheit(100) == 212

def test_absolute_equivalence():
    """Test the temperature where Celsius equals Fahrenheit."""
    assert celsius_to_fahrenheit(-40) == -40

def test_non_numeric_input():
    """Test that non-numeric input raises TypeError."""
    with pytest.raises(TypeError):
        celsius_to_fahrenheit("32")

# Additional useful tests
def test_negative_temperature():
    """Test conversion of negative temperatures."""
    assert celsius_to_fahrenheit(-10) == 14

def test_positive_temperature():
    """Test conversion of positive temperatures."""
    assert celsius_to_fahrenheit(25) == 77
```
