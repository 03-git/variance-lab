I'll show you a comprehensive test suite for the `celsius_to_fahrenheit` function using Python's `unittest` framework.

```python
import unittest

def celsius_to_fahrenheit(c):
    """Convert Celsius to Fahrenheit."""
    return c * 9/5 + 32

class TestCelsiusToFahrenheit(unittest.TestCase):
    
    def test_freezing_point(self):
        """Test conversion at 0°C (freezing point of water)."""
        self.assertEqual(celsius_to_fahrenheit(0), 32)
    
    def test_boiling_point(self):
        """Test conversion at 100°C (boiling point of water)."""
        self.assertEqual(celsius_to_fahrenheit(100), 212)
    
    def test_negative_forty(self):
        """Test conversion at -40°C (where Celsius equals Fahrenheit)."""
        self.assertEqual(celsius_to_fahrenheit(-40), -40)
    
    def test_non_numeric_input(self):
        """Test that non-numeric input raises TypeError."""
        with self.assertRaises(TypeError):
            celsius_to_fahrenheit("100")

if __name__ == '__main__':
    unittest.main()
```
