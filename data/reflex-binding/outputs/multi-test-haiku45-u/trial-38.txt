I'll create a comprehensive Python test suite for the `celsius_to_fahrenheit` function using pytest.

```python
import pytest

def celsius_to_fahrenheit(c):
    return c * 9/5 + 32

class TestCelsiusToFahrenheit:
    def test_freezing_point(self):
        """Test conversion of freezing point (0°C = 32°F)"""
        assert celsius_to_fahrenheit(0) == 32
    
    def test_boiling_point(self):
        """Test conversion of boiling point (100°C = 212°F)"""
        assert celsius_to_fahrenheit(100) == 212
    
    def test_negative_forty(self):
        """Test conversion at -40°C (same in Celsius and Fahrenheit)"""
        assert celsius_to_fahrenheit(-40) == -40
    
    def test_non_numeric_input(self):
        """Test that non-numeric input raises TypeError"""
        with pytest.raises(TypeError):
            celsius_to_fahrenheit("not a number")
    
    def test_non_numeric_input_none(self):
        """Test that None input raises TypeError"""
        with pytest.raises(TypeError):
            celsius_to_fahrenheit(None)
    
    def test_float_input(self):
        """Test conversion with float input (37.5°C ≈ 99.5°F)"""
        assert celsius_to_fahrenheit(37.5) == 99.5
```
