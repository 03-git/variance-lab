```python
import urllib.request
from html.parser import HTMLParser

class LinkExtractor(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []
        self._in_a = False
        self._href = None
        self._text_parts = []

    def handle_starttag(self, tag, attrs):
        if tag == "a":
            self._in_a = True
            self._href = dict(attrs).get("href")
            self._text_parts = []

    def handle_endtag(self, tag):
        if tag == "a" and self._in_a:
            text = "".join(self._text_parts).strip()
            self.links.append((self._href, text))
            self._in_a = False
            self._href = None
            self._text_parts = []

    def handle_data(self, data):
        if self._in_a:
            self._text_parts.append(data)

def main():
    url = "https://example.com"
    req = urllib.request.Request(url, headers={"User-Agent": "Mozilla/5.0"})
    with urllib.request.urlopen(req) as resp:
        html = resp.read().decode(resp.headers.get_content_charset() or "utf-8")

    parser = LinkExtractor()
    parser.feed(html)

    for href, text in parser.links:
        print(f"{href}\t{text}")

if __name__ == "__main__":
    main()
```

How to run:

1. Save as `extract_links.py`.
2. Run `python3 extract_links.py`.

Notes: Uses only the Python standard library (`urllib`, `html.parser`) — no `pip install` needed. If you'd rather use `requests` + `beautifulsoup4`, install them with `pip install requests beautifulsoup4` and I can show that version too.
