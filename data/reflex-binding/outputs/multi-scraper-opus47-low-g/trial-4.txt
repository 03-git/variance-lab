**reflex.1 (python → kernel):** the primitives here are an HTTP GET over a socket and an HTML tokenizer — both in the stdlib (`urllib.request`, `html.parser`). No `requests`, no `beautifulsoup4` needed.

```python
#!/usr/bin/env python3
"""Fetch a URL and print every <a> href with its link text."""
import sys
from html.parser import HTMLParser
from urllib.request import urlopen, Request


class LinkExtractor(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []          # list of [href, text]
        self._href = None        # href of the <a> we're inside, or None
        self._text = []          # text chunks collected inside it

    def handle_starttag(self, tag, attrs):
        if tag == "a":
            href = dict(attrs).get("href")
            if href is not None:
                self._href = href
                self._text = []

    def handle_data(self, data):
        if self._href is not None:
            self._text.append(data)

    def handle_endtag(self, tag):
        if tag == "a" and self._href is not None:
            self.links.append([self._href, "".join(self._text).strip()])
            self._href = None
            self._text = []


def main(url):
    req = Request(url, headers={"User-Agent": "link-extractor/1.0"})
    with urlopen(req, timeout=10) as resp:
        charset = resp.headers.get_content_charset() or "utf-8"
        html = resp.read().decode(charset, errors="replace")

    parser = LinkExtractor()
    parser.feed(html)

    for href, text in parser.links:
        print(f"{href}\t{text!r}")


if __name__ == "__main__":
    main(sys.argv[1] if len(sys.argv) > 1 else "https://example.com")
```

**Run it:**

```bash
python3 link_extractor.py                 # defaults to https://example.com
python3 link_extractor.py https://news.ycombinator.com
```

For `https://example.com` you'll get one line — that page has a single link:

```
https://www.iana.org/domains/example	'More information...'
```

Notes:
- `href` and text are tab-separated; text is `repr()`'d so empty/whitespace-only link text (common for image links) is visible.
- Relative hrefs are printed as-is. If you want them absolute, wrap with `urllib.parse.urljoin(url, href)`.
- No TLS/cert handling added — `urlopen` verifies certs by default, which is what you want.
