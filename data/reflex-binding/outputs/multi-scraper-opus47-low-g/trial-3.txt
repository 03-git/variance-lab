Per reflex.1 — the kernel primitives here are the socket/TLS read (`urllib.request`, stdlib) and a SAX-style parse (`html.parser.HTMLParser`, stdlib). No `requests`, no `bs4` needed.

```python
#!/usr/bin/env python3
import sys
from urllib.request import urlopen, Request
from urllib.parse import urljoin
from html.parser import HTMLParser


class LinkExtractor(HTMLParser):
    def __init__(self, base_url):
        super().__init__()
        self.base_url = base_url
        self.links = []          # list of [href, text]
        self._in_a = False

    def handle_starttag(self, tag, attrs):
        if tag == "a":
            href = dict(attrs).get("href")
            if href is not None:
                self.links.append([urljoin(self.base_url, href), ""])
                self._in_a = True

    def handle_endtag(self, tag):
        if tag == "a":
            self._in_a = False

    def handle_data(self, data):
        if self._in_a and self.links:
            self.links[-1][1] += data.strip()


def main(url="https://example.com"):
    req = Request(url, headers={"User-Agent": "link-extractor/1.0"})
    with urlopen(req, timeout=10) as resp:
        charset = resp.headers.get_content_charset() or "utf-8"
        html = resp.read().decode(charset, errors="replace")

    parser = LinkExtractor(url)
    parser.feed(html)

    if not parser.links:
        print("(no anchor tags found)")
    for href, text in parser.links:
        print(f"{href}\t{text or '(no text)'}")


if __name__ == "__main__":
    main(sys.argv[1] if len(sys.argv) > 1 else "https://example.com")
```

Run it:

```bash
python3 script.py                    # defaults to https://example.com
python3 script.py https://some.other/page   # optional: pass any URL
```

Notes:
- `urljoin` resolves relative hrefs against the page URL so you get absolute links.
- `example.com` has exactly one anchor (the IANA "More information..." link), so expect one line of output — it's a thin test page, not a bug.
- Pure stdlib: nothing to `pip install`.
