```python
#!/usr/bin/env python3
import argparse
import json
import sys


def port_type(value):
    try:
        port = int(value)
    except ValueError:
        raise argparse.ArgumentTypeError(f"port must be an integer, got {value!r}")
    if not 1 <= port <= 65535:
        raise argparse.ArgumentTypeError(f"port must be between 1 and 65535, got {port}")
    return port


def main():
    parser = argparse.ArgumentParser(description="Print parsed config as JSON.")
    parser.add_argument("--host", required=True)
    parser.add_argument("--port", required=True, type=port_type)
    parser.add_argument("--verbose", action="store_true")
    args = parser.parse_args()

    config = {"host": args.host, "port": args.port, "verbose": args.verbose}
    json.dump(config, sys.stdout, indent=2)
    sys.stdout.write("\n")


if __name__ == "__main__":
    main()
```

Run it:

```
python3 config_cli.py --host 127.0.0.1 --port 8080 --verbose
```

Invalid port exits non-zero with an argparse error:

```
python3 config_cli.py --host 127.0.0.1 --port 70000
```
