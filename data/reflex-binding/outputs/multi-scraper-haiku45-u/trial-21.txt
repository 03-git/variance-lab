I'll show you a Python script that fetches a URL and extracts hyperlinks. Here's the code:

```python
import requests
from html.parser import HTMLParser
from urllib.parse import urljoin

class LinkExtractor(HTMLParser):
    def __init__(self, base_url):
        super().__init__()
        self.base_url = base_url
        self.links = []
        self.current_text = []
        self.in_anchor = False
    
    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            self.in_anchor = True
            self.current_text = []
            attrs_dict = dict(attrs)
            href = attrs_dict.get('href', '')
            # Convert relative URLs to absolute
            absolute_url = urljoin(self.base_url, href)
            self.links.append({'href': absolute_url, 'text': ''})
    
    def handle_endtag(self, tag):
        if tag == 'a':
            self.in_anchor = False
            if self.links:
                self.links[-1]['text'] = ''.join(self.current_text).strip()
    
    def handle_data(self, data):
        if self.in_anchor:
            self.current_text.append(data)

def fetch_and_extract_links(url):
    try:
        response = requests.get(url, timeout=10)
        response.raise_for_status()
        
        parser = LinkExtractor(url)
        parser.feed(response.text)
        
        return parser.links
    except requests.RequestException as e:
        print(f"Error fetching {url}: {e}")
        return []

if __name__ == '__main__':
    url = 'https://example.com'
    links = fetch_and_extract_links(url)
    
    print(f"Found {len(links)} links:\n")
    for i, link in enumerate(links, 1):
        href = link['href']
        text = link['text'] or '(no text)'
        print(f"{i}. {text}")
        print(f"   URL: {href}\n")
```

**How to run it:**

1. **Install dependencies** (if you don't have them):
   ```bash
   pip install requests
   ```

2. **Save the script** to a file (e.g., `extract_links.py`) and run:
   ```bash
   python extract_links.py
   ```

**What it does:**
- Fetches the HTML from `https://example.com`
- Uses `HTMLParser` to extract all `<a>` tags
- Captures the `href` attribute and the link text
- Converts relative URLs to absolute URLs using `urljoin`
- Prints each link with its text and full URL

**Alternative using BeautifulSoup** (cleaner, if you prefer):
```python
import requests
from bs4 import BeautifulSoup
from urllib.parse import urljoin

url = 'https://example.com'
response = requests.get(url, timeout=10)
response.raise_for_status()

soup = BeautifulSoup(response.text, 'html.parser')
links = soup.find_all('a')

print(f"Found {len(links)} links:\n")
for i, link in enumerate(links, 1):
    href = urljoin(url, link.get('href', ''))
    text = link.get_text(strip=True) or '(no text)'
    print(f"{i}. {text}")
    print(f"   URL: {href}\n")
```

Install BeautifulSoup with: `pip install beautifulsoup4`

Both approaches work — the first uses only the standard library + requests, while the second is more concise with BeautifulSoup.
