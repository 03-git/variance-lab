I'll show you a Python script that fetches a URL and extracts hyperlinks:

```python
import requests
from html.parser import HTMLParser
from urllib.parse import urljoin

class LinkExtractor(HTMLParser):
    def __init__(self, base_url):
        super().__init__()
        self.base_url = base_url
        self.links = []
        self.current_text = []
    
    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            attrs_dict = dict(attrs)
            href = attrs_dict.get('href', '')
            # Convert relative URLs to absolute
            if href:
                href = urljoin(self.base_url, href)
            self.current_text = []
    
    def handle_data(self, data):
        self.current_text.append(data.strip())
    
    def handle_endtag(self, tag):
        if tag == 'a':
            text = ' '.join(self.current_text).strip()
            if self.links and not self.links[-1][0]:  # Get the href from current anchor
                href = self.links[-1][0] if self.links else ''
            # Store the link
            if hasattr(self, '_current_href'):
                self.links.append((self._current_href, text))
                delattr(self, '_current_href')
    
    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            attrs_dict = dict(attrs)
            href = attrs_dict.get('href', '')
            if href:
                href = urljoin(self.base_url, href)
            self._current_href = href
            self.current_text = []
    
    def handle_data(self, data):
        self.current_text.append(data.strip())
    
    def handle_endtag(self, tag):
        if tag == 'a' and hasattr(self, '_current_href'):
            text = ' '.join(self.current_text).strip()
            self.links.append((self._current_href, text))

def fetch_and_extract_links(url):
    try:
        response = requests.get(url, timeout=10)
        response.raise_for_status()
    except requests.RequestException as e:
        print(f"Error fetching {url}: {e}")
        return
    
    parser = LinkExtractor(url)
    parser.feed(response.text)
    
    print(f"Found {len(parser.links)} links:\n")
    for href, text in parser.links:
        print(f"URL: {href}")
        print(f"Text: {text if text else '(empty)'}")
        print()

if __name__ == "__main__":
    fetch_and_extract_links("https://example.com")
```

**How to run it:**

1. **Install the required library:**
   ```bash
   pip install requests
   ```

2. **Run the script:**
   ```bash
   python script.py
   ```

**What it does:**
- Uses `requests` to fetch the webpage
- Uses `HTMLParser` to parse the HTML and extract `<a>` tags
- Converts relative URLs to absolute URLs using `urljoin()`
- Prints each link's href and the link text

**Output example:**
```

URL: https://www.iana.org/domains/example
Text: More information...

URL: https://www.iana.org/
Text: IANA
...
```

You can modify the URL by changing `"https://example.com"` to any other site you want to scrape.
