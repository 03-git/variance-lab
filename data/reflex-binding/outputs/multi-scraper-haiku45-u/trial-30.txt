I'll show you a Python script that fetches a webpage and extracts hyperlinks:

```python
import requests
from html.parser import HTMLParser
from urllib.parse import urljoin

class LinkExtractor(HTMLParser):
    def __init__(self, base_url):
        super().__init__()
        self.base_url = base_url
        self.links = []
        self.current_text = []
        self.in_anchor = False

    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            self.in_anchor = True
            self.current_text = []
            for attr, value in attrs:
                if attr == 'href':
                    # Resolve relative URLs
                    absolute_url = urljoin(self.base_url, value)
                    self.links.append({'href': absolute_url, 'text': ''})

    def handle_data(self, data):
        if self.in_anchor:
            self.current_text.append(data.strip())

    def handle_endtag(self, tag):
        if tag == 'a' and self.in_anchor:
            self.in_anchor = False
            if self.links:
                self.links[-1]['text'] = ' '.join(self.current_text)

def fetch_and_extract_links(url):
    try:
        response = requests.get(url, timeout=10)
        response.raise_for_status()
        
        parser = LinkExtractor(url)
        parser.feed(response.text)
        
        return parser.links
    except requests.RequestException as e:
        print(f"Error fetching {url}: {e}")
        return []

if __name__ == '__main__':
    url = 'https://example.com'
    links = fetch_and_extract_links(url)
    
    if links:
        print(f"Found {len(links)} links:\n")
        for i, link in enumerate(links, 1):
            href = link['href']
            text = link['text'] or '(no text)'
            print(f"{i}. {text}")
            print(f"   → {href}\n")
    else:
        print("No links found or failed to fetch the page.")
```

**How to run it:**

1. Install the required package:
```bash
pip install requests
```

2. Run the script:
```bash
python script.py
```

**What it does:**
- Fetches the HTML from `https://example.com`
- Uses `HTMLParser` to extract all anchor tags
- Captures both the `href` attribute and the link text
- Resolves relative URLs to absolute ones using `urljoin`
- Prints each link with its text and full URL

**Alternative with BeautifulSoup** (simpler approach):
```python
import requests
from bs4 import BeautifulSoup
from urllib.parse import urljoin

url = 'https://example.com'
response = requests.get(url, timeout=10)
response.raise_for_status()

soup = BeautifulSoup(response.text, 'html.parser')

links = soup.find_all('a')
print(f"Found {len(links)} links:\n")

for i, link in enumerate(links, 1):
    href = urljoin(url, link.get('href', ''))
    text = link.get_text(strip=True) or '(no text)'
    print(f"{i}. {text}")
    print(f"   → {href}\n")
```

Install with: `pip install requests beautifulsoup4`

The BeautifulSoup version is cleaner if you prefer brevity. The first version uses only the standard library (besides `requests`).
