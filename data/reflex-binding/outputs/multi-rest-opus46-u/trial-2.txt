Here's a compact implementation using **Flask** (Python).

```python
# app.py
from flask import Flask, request, jsonify

app = Flask(__name__)


def validate_user(data):
    errors = {}

    if not isinstance(data, dict):
        return None, {"_": "request body must be a JSON object"}

    # name: string, required
    name = data.get("name")
    if name is None:
        errors["name"] = "required"
    elif not isinstance(name, str) or not name.strip():
        errors["name"] = "must be a non-empty string"

    # email: string, must contain '@'
    email = data.get("email")
    if email is None:
        errors["email"] = "required"
    elif not isinstance(email, str) or "@" not in email:
        errors["email"] = "must be a string containing '@'"

    # age: integer, 1-150
    age = data.get("age")
    if age is None:
        errors["age"] = "required"
    elif isinstance(age, bool) or not isinstance(age, int):
        errors["age"] = "must be an integer"
    elif not (1 <= age <= 150):
        errors["age"] = "must be between 1 and 150"

    if errors:
        return None, errors

    return {"name": name, "email": email, "age": age}, None


@app.post("/users")
def create_user():
    data = request.get_json(silent=True)
    user, errors = validate_user(data)
    if errors:
        return jsonify({"error": "invalid input", "details": errors}), 400
    return jsonify(user), 201


if __name__ == "__main__":
    app.run(port=8000, debug=True)
```

### How to run

```bash
pip install flask
python app.py
```

The server listens on `http://localhost:8000`.

### Try it

```bash
# success -> 201
curl -i -X POST http://localhost:8000/users \
  -H 'Content-Type: application/json' \
  -d '{"name":"Ada","email":"ada@example.com","age":36}'

# invalid -> 400 with details
curl -i -X POST http://localhost:8000/users \
  -H 'Content-Type: application/json' \
  -d '{"name":"","email":"nope","age":999}'
```

Expected `400` body for the second request:

```json
{
  "error": "invalid input",
  "details": {
    "name": "must be a non-empty string",
    "email": "must be a string containing '@'",
    "age": "must be between 1 and 150"
  }
}
```

Notes:
- `request.get_json(silent=True)` returns `None` on missing/invalid JSON, which falls through to a `400`.
- The `isinstance(age, bool)` check is deliberate — in Python `True`/`False` are instances of `int`, and we don't want `"age": true` to pass.
