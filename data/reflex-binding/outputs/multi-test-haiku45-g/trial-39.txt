I'll show you a comprehensive Python test suite for that function using `unittest`, the standard testing framework.

```python
import unittest

def celsius_to_fahrenheit(c):
    """Convert temperature from Celsius to Fahrenheit."""
    return c * 9/5 + 32

class TestCelsiusToFahrenheit(unittest.TestCase):
    
    def test_zero_celsius(self):
        """0°C should equal 32°F (freezing point of water)."""
        self.assertEqual(celsius_to_fahrenheit(0), 32)
    
    def test_boiling_point(self):
        """100°C should equal 212°F (boiling point of water)."""
        self.assertEqual(celsius_to_fahrenheit(100), 212)
    
    def test_negative_forty(self):
        """-40°C should equal -40°F (same in both scales)."""
        self.assertEqual(celsius_to_fahrenheit(-40), -40)
    
    def test_non_numeric_input(self):
        """Non-numeric input should raise TypeError."""
        with self.assertRaises(TypeError):
            celsius_to_fahrenheit("not a number")

if __name__ == '__main__':
    unittest.main()
```
