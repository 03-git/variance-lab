```python
import urllib.request
from html.parser import HTMLParser

class LinkParser(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []
        self._current_href = None
        self._current_text = []

    def handle_starttag(self, tag, attrs):
        if tag == "a":
            attrs = dict(attrs)
            self._current_href = attrs.get("href")
            self._current_text = []

    def handle_data(self, data):
        if self._current_href is not None:
            self._current_text.append(data)

    def handle_endtag(self, tag):
        if tag == "a" and self._current_href is not None:
            self.links.append((self._current_href, "".join(self._current_text).strip()))
            self._current_href = None
            self._current_text = []

with urllib.request.urlopen("https://example.com") as resp:
    html = resp.read().decode("utf-8")

parser = LinkParser()
parser.feed(html)

for href, text in parser.links:
    print(f"{text!r:30} -> {href}")
```

**Run it:**

```bash
python3 script.py
```

No third-party dependencies — uses only stdlib (`urllib.request` and `html.parser`). If you want to extend this to arbitrary sites with relative URL resolution, `urllib.parse.urljoin` can normalize hrefs against the base URL.
