```python
from http.server import BaseHTTPRequestHandler, HTTPServer
import json, datetime

class H(BaseHTTPRequestHandler):
    def do_GET(self):
        if self.path == "/time":
            self.send_response(200)
            self.send_header("Content-Type", "application/json")
            self.end_headers()
            self.wfile.write(json.dumps({"time": datetime.datetime.now(datetime.timezone.utc).isoformat()}).encode())
        else:
            self.send_response(404)
            self.end_headers()

HTTPServer(("", 8000), H).serve_forever()
```

Run: `python3 server.py` — serves on port 8000. Test: `curl localhost:8000/time`.
