I'll show you a Python script that fetches a URL and extracts hyperlinks:

```python
import requests
from html.parser import HTMLParser
from urllib.parse import urljoin

class LinkExtractor(HTMLParser):
    def __init__(self, base_url):
        super().__init__()
        self.base_url = base_url
        self.links = []
        self.current_text = []
    
    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            attrs_dict = dict(attrs)
            href = attrs_dict.get('href', '')
            # Convert relative URLs to absolute
            absolute_url = urljoin(self.base_url, href)
            self.links.append({'href': absolute_url, 'text': ''})
    
    def handle_data(self, data):
        if self.links:
            self.links[-1]['text'] += data.strip()
    
    def handle_endtag(self, tag):
        if tag == 'a' and self.links:
            self.links[-1]['text'] = self.links[-1]['text'].strip()

def fetch_and_extract_links(url):
    try:
        response = requests.get(url, timeout=10)
        response.raise_for_status()
        
        parser = LinkExtractor(url)
        parser.feed(response.text)
        
        print(f"Found {len(parser.links)} links:\n")
        for link in parser.links:
            text = link['text'] or "(no text)"
            print(f"Text: {text}")
            print(f"URL:  {link['href']}\n")
    
    except requests.RequestException as e:
        print(f"Error fetching {url}: {e}")

if __name__ == "__main__":
    fetch_and_extract_links("https://example.com")
```

2. Save the script as `fetch_links.py` and run:
   ```bash
   python fetch_links.py
   ```
