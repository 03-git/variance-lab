```python
import urllib.request
from html.parser import HTMLParser

class LinkExtractor(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []
        self.current_href = None
        self.current_text = ""

    def handle_starttag(self, tag, attrs):
        if tag == "a":
            for name, value in attrs:
                if name == "href":
                    self.current_href = value
                    self.current_text = ""

    def handle_data(self, data):
        if self.current_href is not None:
            self.current_text += data

    def handle_endtag(self, tag):
        if tag == "a" and self.current_href is not None:
            self.links.append((self.current_href, self.current_text.strip()))
            self.current_href = None
            self.current_text = ""

response = urllib.request.urlopen("https://example.com")
html = response.read().decode()

parser = LinkExtractor()
parser.feed(html)

for href, text in parser.links:
    print(f"{href} -> {text}")
```

Run it with:

```
python3 script.py
```

No dependencies beyond the standard library. Uses `urllib.request` to fetch the page and `html.parser.HTMLParser` to walk the DOM and pull out `<a href="...">text</a>` pairs.
