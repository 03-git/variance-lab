```python
#!/usr/bin/env python3
"""Fetch a URL and print all hyperlinks (href + link text)."""

import sys
from urllib.request import urlopen, Request
from html.parser import HTMLParser


class LinkParser(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []          # list of (href, text)
        self._href = None
        self._text_parts = []

    def handle_starttag(self, tag, attrs):
        if tag == "a":
            href = dict(attrs).get("href")
            if href is not None:
                self._href = href
                self._text_parts = []

    def handle_data(self, data):
        if self._href is not None:
            self._text_parts.append(data)

    def handle_endtag(self, tag):
        if tag == "a" and self._href is not None:
            text = " ".join("".join(self._text_parts).split())
            self.links.append((self._href, text))
            self._href = None
            self._text_parts = []


def main():
    url = sys.argv[1] if len(sys.argv) > 1 else "https://example.com"
    req = Request(url, headers={"User-Agent": "Mozilla/5.0 (link-extractor)"})
    with urlopen(req, timeout=10) as resp:
        charset = resp.headers.get_content_charset() or "utf-8"
        html = resp.read().decode(charset, errors="replace")

    parser = LinkParser()
    parser.feed(html)

    if not parser.links:
        print("No links found.")
        return

    for href, text in parser.links:
        print(f"{href}\t{text or '(no text)'}")


if __name__ == "__main__":
    main()
```

How to run it:

```bash
# Save the code to a file, e.g. links.py, then:
python3 links.py                  # defaults to https://example.com
python3 links.py https://news.ycombinator.com   # or pass any URL
```

Notes:
- Uses only the Python standard library (`urllib`, `html.parser`) — no `pip install` needed.
- For `https://example.com` you'll get one link: `https://www.iana.org/domains/example` with text `More information...`.
- If you'd rather use `requests` + `beautifulsoup4` (more robust for messy real-world HTML), say the word and I'll show that version.
