```python
#!/usr/bin/env python3
"""Fetch a URL and print every <a> tag's href and link text."""

import sys
from html.parser import HTMLParser
from urllib.request import urlopen, Request


class LinkExtractor(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []          # list of [href, text]
        self._current_href = None
        self._current_text = []

    def handle_starttag(self, tag, attrs):
        if tag == "a":
            href = dict(attrs).get("href")
            if href is not None:
                self._current_href = href
                self._current_text = []

    def handle_data(self, data):
        if self._current_href is not None:
            self._current_text.append(data)

    def handle_endtag(self, tag):
        if tag == "a" and self._current_href is not None:
            text = " ".join("".join(self._current_text).split())
            self.links.append([self._current_href, text])
            self._current_href = None
            self._current_text = []


def main(url="https://example.com"):
    req = Request(url, headers={"User-Agent": "link-extractor/1.0"})
    with urlopen(req) as resp:
        charset = resp.headers.get_content_charset() or "utf-8"
        html = resp.read().decode(charset, errors="replace")

    parser = LinkExtractor()
    parser.feed(html)

    if not parser.links:
        print("No links found.")
        return

    for href, text in parser.links:
        print(f"{href}\t{text or '(no text)'}")


if __name__ == "__main__":
    main(sys.argv[1] if len(sys.argv) > 1 else "https://example.com")
```

**How to run it**

1. Save the code as `extract_links.py`.
2. Run with the default URL (`https://example.com`):
   ```
   python3 extract_links.py
   ```
3. Or pass any other URL:
   ```
   python3 extract_links.py https://www.python.org
   ```

Notes:
- Uses only the Python standard library (`urllib`, `html.parser`) — no `pip install` needed.
- Output is one link per line: `href` then a tab then the link text.
- For `https://example.com` you'll get a single link to `https://www.iana.org/domains/example` with text "More information...".
- If you'd prefer a version using `requests` + `BeautifulSoup` (more robust on messy HTML), let me know.
