I'll show you a comprehensive Python test suite using pytest. Here's the code:

```python
def celsius_to_fahrenheit(c):
    """Convert temperature from Celsius to Fahrenheit."""
    return c * 9/5 + 32


import pytest

class TestCelsiusToFahrenheit:
    """Test suite for celsius_to_fahrenheit function."""
    
    def test_freezing_point(self):
        """Test conversion at freezing point (0°C = 32°F)."""
        assert celsius_to_fahrenheit(0) == 32
    
    def test_boiling_point(self):
        """Test conversion at boiling point (100°C = 212°F)."""
        assert celsius_to_fahrenheit(100) == 212
    
    def test_same_temperature(self):
        """Test conversion at -40 where Celsius equals Fahrenheit."""
        assert celsius_to_fahrenheit(-40) == -40
    
    def test_non_numeric_input(self):
        """Test that non-numeric input raises TypeError."""
        with pytest.raises(TypeError):
            celsius_to_fahrenheit("not a number")
    
    def test_negative_temperature(self):
        """Test conversion for negative temperature (bonus test)."""
        assert celsius_to_fahrenheit(-10) == 14


# Alternative: unittest version (no external dependency)
import unittest

class TestCelsiusToFahrenheitUnittest(unittest.TestCase):
    """Test suite using unittest (standard library)."""
    
    def test_freezing_point(self):
        self.assertEqual(celsius_to_fahrenheit(0), 32)
    
    def test_boiling_point(self):
        self.assertEqual(celsius_to_fahrenheit(100), 212)
    
    def test_same_temperature(self):
        self.assertEqual(celsius_to_fahrenheit(-40), -40)
    
    def test_non_numeric_input(self):
        with self.assertRaises(TypeError):
            celsius_to_fahrenheit("not a number")
```

**How to run:**

**With pytest (recommended):**
```bash
# Install pytest if needed
pip install pytest

# Run the tests
pytest test_file.py -v

# Run a specific test
pytest test_file.py::TestCelsiusToFahrenheit::test_freezing_point -v
```

**With unittest (no install needed):**
```bash
# Run all tests
python -m unittest test_file.py -v

# Run specific test class
python -m unittest test_file.py.TestCelsiusToFahrenheitUnittest -v
```
