```python
#!/usr/bin/env python3
import sqlite3
from datetime import datetime, timezone

DB = "notes.db"

def main():
    conn = sqlite3.connect(DB)
    cur = conn.cursor()
    cur.execute("""
        CREATE TABLE IF NOT EXISTS notes (
            id INTEGER PRIMARY KEY,
            title TEXT,
            body TEXT,
            created_at TEXT
        )
    """)

    now = datetime.now(timezone.utc).isoformat()
    samples = [
        ("Groceries", "Milk, eggs, coffee", now),
        ("Idea", "Local inference daemon for the kids' tablet", now),
        ("Reminder", "Sign the session manifest", now),
    ]
    cur.executemany(
        "INSERT INTO notes (title, body, created_at) VALUES (?, ?, ?)",
        samples,
    )
    conn.commit()

    for row in cur.execute("SELECT id, title, body, created_at FROM notes ORDER BY id"):
        print(row)

    conn.close()

if __name__ == "__main__":
    main()
```

Run it:

```bash
python3 notes_demo.py
```

Notes:
- `sqlite3` is in the Python standard library — no install needed.
- It creates `notes.db` in the current directory. Re-running appends 3 more rows each time (the `CREATE TABLE IF NOT EXISTS` won't clear it); `rm notes.db` to start fresh.
- Expected output is three tuples like `(1, 'Groceries', 'Milk, eggs, coffee', '2026-05-10T...+00:00')`.
