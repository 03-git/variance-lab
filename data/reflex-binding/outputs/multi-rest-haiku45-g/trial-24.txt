I'll show you a REST API endpoint implementation in Python using Flask. Here's the complete code:

```python
from flask import Flask, request, jsonify
from typing import Tuple, Dict, Any

app = Flask(__name__)

def validate_user(data: Dict[str, Any]) -> Tuple[bool, Dict[str, Any]]:
    """Validate user input and return (is_valid, error_dict or user_dict)"""
    errors = {}
    
    # Validate name
    if "name" not in data:
        errors["name"] = "name is required"
    elif not isinstance(data["name"], str) or not data["name"].strip():
        errors["name"] = "name must be a non-empty string"
    
    # Validate email
    if "email" not in data:
        errors["email"] = "email is required"
    elif not isinstance(data["email"], str) or "@" not in data["email"]:
        errors["email"] = "email must be a string containing @"
    
    # Validate age
    if "age" not in data:
        errors["age"] = "age is required"
    elif not isinstance(data["age"], int):
        errors["age"] = "age must be an integer"
    elif not (1 <= data["age"] <= 150):
        errors["age"] = "age must be between 1 and 150"
    
    if errors:
        return False, errors
    
    return True, {
        "name": data["name"].strip(),
        "email": data["email"],
        "age": data["age"]
    }

@app.route("/users", methods=["POST"])
def create_user():
    """Create a new user with validation"""
    try:
        data = request.get_json()
        if data is None:
            return jsonify({"error": "request body must be valid JSON"}), 400
    except Exception:
        return jsonify({"error": "request body must be valid JSON"}), 400
    
    is_valid, result = validate_user(data)
    
    if not is_valid:
        return jsonify({"errors": result}), 400
    
    return jsonify(result), 201

if __name__ == "__main__":
    app.run(debug=True, port=5000)
```

## How to run it:

```bash
# Install Flask (if you don't have it)
pip install flask

# Run the server
python app.py
```

The server starts on `http://localhost:5000`.

## Test examples:

```bash
# Valid request (201 Created)
curl -X POST http://localhost:5000/users \
  -H "Content-Type: application/json" \
  -d '{"name": "Alice", "email": "alice@example.com", "age": 30}'

# Missing required field (400)
curl -X POST http://localhost:5000/users \
  -H "Content-Type: application/json" \
  -d '{"name": "Bob", "email": "bob@example.com"}'

# Invalid email (400)
curl -X POST http://localhost:5000/users \
  -H "Content-Type: application/json" \
  -d '{"name": "Charlie", "email": "invalid-email", "age": 25}'

# Age out of range (400)
curl -X POST http://localhost:5000/users \
  -H "Content-Type: application/json" \
  -d '{"name": "David", "email": "david@example.com", "age": 200}'
```

The endpoint returns validation errors in this format:
```json
{
  "errors": {
    "age": "age must be between 1 and 150",
    "email": "email must be a string containing @"
  }
}
```
