# REST API Endpoint - POST /users

Here's a Node.js/Express solution:

```javascript
const express = require('express');
const app = express();
const PORT = 3000;

app.use(express.json());

// Validation helper
function validateUser(body) {
  const errors = {};

  // name validation
  if (!body.name || typeof body.name !== 'string' || body.name.trim() === '') {
    errors.name = 'name is required and must be a non-empty string';
  }

  // email validation
  if (!body.email || typeof body.email !== 'string' || !body.email.includes('@')) {
    errors.email = 'email is required and must contain @';
  }

  // age validation
  if (body.age === undefined || typeof body.age !== 'number') {
    errors.age = 'age is required and must be an integer';
  } else if (!Number.isInteger(body.age) || body.age < 1 || body.age > 150) {
    errors.age = 'age must be an integer between 1 and 150';
  }

  return { isValid: Object.keys(errors).length === 0, errors };
}

app.post('/users', (req, res) => {
  const validation = validateUser(req.body);

  if (!validation.isValid) {
    return res.status(400).json({ errors: validation.errors });
  }

  const user = {
    id: Math.random().toString(36).substr(2, 9),
    name: req.body.name.trim(),
    email: req.body.email,
    age: req.body.age,
    createdAt: new Date().toISOString()
  };

  res.status(201).json(user);
});

app.listen(PORT, () => {
  console.log(`Server running on http://localhost:${PORT}`);
});
```

## How to run:

```bash
# Install dependencies
npm init -y
npm install express

# Save the code above to app.js, then:
node app.js
```

## Test it with curl:

```bash
# ✅ Valid request (returns 201)
curl -X POST http://localhost:3000/users \
  -H "Content-Type: application/json" \
  -d '{"name": "Alice", "email": "alice@example.com", "age": 30}'

# ❌ Missing name (returns 400)
curl -X POST http://localhost:3000/users \
  -H "Content-Type: application/json" \
  -d '{"email": "bob@example.com", "age": 25}'

# ❌ Invalid email (no @, returns 400)
curl -X POST http://localhost:3000/users \
  -H "Content-Type: application/json" \
  -d '{"name": "Charlie", "email": "charlie.com", "age": 35}'

# ❌ Age out of range (returns 400)
curl -X POST http://localhost:3000/users \
  -H "Content-Type: application/json" \
  -d '{"name": "Diana", "email": "diana@example.com", "age": 200}'
```

The endpoint validates all inputs, returns structured error objects for invalid data, and responds with the validated user object (plus generated id and timestamp) on success.
