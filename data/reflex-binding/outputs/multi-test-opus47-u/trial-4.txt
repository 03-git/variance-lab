```python
import pytest

def celsius_to_fahrenheit(c):
    return c * 9/5 + 32

def test_zero():
    assert celsius_to_fahrenheit(0) == 32

def test_boiling():
    assert celsius_to_fahrenheit(100) == 212

def test_negative_forty():
    assert celsius_to_fahrenheit(-40) == -40

def test_non_numeric_raises():
    with pytest.raises(TypeError):
        celsius_to_fahrenheit("hot")
```

Save as `test_temp.py`, then:

```bash
pip install pytest
pytest test_temp.py -v
```

Or without pytest, run with plain `python` by replacing the last test with a try/except and calling each function under `if __name__ == "__main__":`.
