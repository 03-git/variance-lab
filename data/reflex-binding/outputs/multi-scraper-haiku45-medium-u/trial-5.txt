I'll show you a Python script that fetches a URL and extracts hyperlinks from the HTML.

```python
import requests
from html.parser import HTMLParser

class LinkExtractor(HTMLParser):
    def __init__(self):
        super().__init__()
        self.links = []
        self.current_text = []
    
    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            attrs_dict = dict(attrs)
            href = attrs_dict.get('href', '')
            self.current_text = []
    
    def handle_data(self, data):
        self.current_text.append(data.strip())
    
    def handle_endtag(self, tag):
        if tag == 'a':
            text = ' '.join(self.current_text)
            if self.links and not self.links[-1][1]:
                self.links[-1] = (self.links[-1][0], text)
    
    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            attrs_dict = dict(attrs)
            href = attrs_dict.get('href', '')
            self.links.append((href, ''))
        super().handle_starttag(tag, attrs)
    
    def handle_data(self, data):
        if self.links:
            href, text = self.links[-1]
            self.links[-1] = (href, text + data.strip())
        super().handle_data(data)

def fetch_and_extract_links(url):
    try:
        response = requests.get(url, timeout=10)
        response.raise_for_status()
        
        parser = LinkExtractor()
        parser.feed(response.text)
        
        return parser.links
    except requests.RequestException as e:
        print(f"Error fetching {url}: {e}")
        return []

if __name__ == '__main__':
    url = 'https://example.com'
    links = fetch_and_extract_links(url)
    
    if links:
        print(f"Found {len(links)} links:\n")
        for href, text in links:
            print(f"URL: {href}")
            print(f"Text: {text if text else '(no text)'}")
            print()
    else:
        print("No links found.")
```

**How to run it:**

1. Install the required dependency:
```bash
pip install requests
```

2. Run the script:
```bash
python script.py
```

The script will:
- Fetch the HTML from the URL
- Parse all `<a>` tags
- Extract the `href` attribute and the link text
- Print each link's URL and text (or "(no text)" if the anchor has no text content)

You can modify the `url` variable on line that sets it to fetch any website you want to analyze.
