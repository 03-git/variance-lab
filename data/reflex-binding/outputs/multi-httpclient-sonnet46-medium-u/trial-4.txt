```python
import sys
import urllib.request

def fetch(url):
    with urllib.request.urlopen(url) as response:
        print(f"Status: {response.status}")
        print("\nHeaders:")
        for key, value in response.headers.items():
            print(f"  {key}: {value}")
        body = response.read(500).decode("utf-8", errors="replace")
        print(f"\nBody (first 500 chars):\n{body}")

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("Usage: python fetch.py <url>")
        sys.exit(1)
    fetch(sys.argv[1])
```

`urllib.request.urlopen` follows redirects automatically. No third-party dependencies needed.

**How to run:**

```bash
python fetch.py https://example.com
```

If you prefer using `requests` (more readable headers, explicit redirect control):

```python
import sys
import requests

def fetch(url):
    response = requests.get(url, allow_redirects=True)
    print(f"Status: {response.status_code}")
    print("\nHeaders:")
    for key, value in response.headers.items():
        print(f"  {key}: {value}")
    print(f"\nBody (first 500 chars):\n{response.text[:500]}")

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("Usage: python fetch.py <url>")
        sys.exit(1)
    fetch(sys.argv[1])
```
