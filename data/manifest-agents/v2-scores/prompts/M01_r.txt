__subtract_lookup_file() {
    local input_lower
    input_lower=$(__subtract_lower "$1")
    local pattern tag cmd rest pattern_lower
    while IFS=$'\t' read -r pattern rest; do
        [[ "$pattern" =~ ^#.*$ || -z "$pattern" ]] && continue
        pattern_lower=$(__subtract_lower "$pattern")
        # shellcheck disable=SC2254
        if [ -n "$ZSH_VERSION" ]; then
            [[ "$input_lower" == $~pattern_lower ]] || continue
        else
            [[ "$input_lower" == $pattern_lower ]] || continue
        fi
        if [[ "$rest" =~ ^\[([a-z]+)\] ]]; then
            if [ -n "$ZSH_VERSION" ]; then tag="${match[1]}"; else tag="${BASH_REMATCH[1]}"; fi
            cmd="${rest#*	}"
        else
            tag="stdout"
            cmd="$rest"
        fi
        echo "${tag}	${cmd}"

static int lookdown_check(const char *input, char *output, int output_max,
                          int *exec_out, int *match_idx) {
    char lower[4096];
    int i;
    for (i = 0; input[i] && i < (int)sizeof(lower) - 1; i++)
        lower[i] = (input[i] >= 'A' && input[i] <= 'Z') ? input[i] + 32 : input[i];
    lower[i] = '\0';
    if (exec_out) *exec_out = 0;
    if (match_idx) *match_idx = -1;

    for (int j = 0; j < lookdown_count; j++) {
        if (lookdown_table[j].flags & LOOKDOWN_EXEC) continue;
        if (fnmatch(lookdown_table[j].pattern, lower, 0) == 0) {
            if (match_idx) *match_idx = j;
            FILE *fp = popen(lookdown_table[j].command, "r");
            if (!fp) return -1;
            int len = fread(output, 1, output_max - 1, fp);
            output[len] = '\0';
            pclose(fp);
            while (len > 0 && output[len - 1] == '\n') output[--len] = '\0';
            return len;
        }
    }
    return -1;
}

check_lookdown() {
    local input; input=$(printf '%s' "$1" | tr '[:upper:]' '[:lower:]')
    local line pattern cmd
    for file in "$LOOKDOWN_NODE" "$LOOKDOWN_UNIVERSAL"; do
        [ -f "$file" ] || continue
        while IFS=$'\t' read -r pattern cmd rest; do
            [ -z "$pattern" ] && continue
            [[ "$pattern" == \#* ]] && continue
            # glob match
            case "$input" in
                $pattern) echo "$cmd"; return 0 ;;
            esac
        done < "$file"
    done
    return 1
}

static int run_entry_visible(hosuni_app *app, int i) {
    if (!(lookdown_table[i].flags & LOOKDOWN_EXEC)) return 0;
    if (!run_match_filter(lookdown_table[i].pattern, app->run.filter, app->run.filter_len)) return 0;
    if (!app->run.show_all && has_any_pinned() && !(lookdown_table[i].flags & LOOKDOWN_PINNED)) return 0;
    return 1;
}

static int run_entry_count(hosuni_app *app) {
    int n = 0;
    for (int i = 0; i < lookdown_count; i++)
        if (run_entry_visible(app, i)) n++;
    return n;
}

static int run_entry_index(hosuni_app *app, int nth) {
    int n = 0;
    for (int i = 0; i < lookdown_count; i++) {
        if (run_entry_visible(app, i)) {
            if (n == nth) return i;
            n++;
        }
    }
    return -1;
}

## Finding

A lookdown table is a dataset. The QLoRA-portability test
("would this improve any user's local model?") is a clean
design constraint that separates infrastructure from taste.
The glob pattern discipline ("match what you authored for,
not everything containing that word") prevents the lookup
tier from starving the inference tier.

The distinction between Qwen (tool calls, structured output)
and Gemma-4 (knowledge questions, does not confabulate) is
an empirical routing decision, not an architecture one. The
same `cal_infer_stream_msgs` code path serves both — only
the host and port differ.
