static int lookdown_check(const char *input, char *output, int output_max,
                          int *exec_out, int *match_idx) {
    char lower[4096];
    int i;
    for (i = 0; input[i] && i < (int)sizeof(lower) - 1; i++)
        lower[i] = (input[i] >= 'A' && input[i] <= 'Z') ? input[i] + 32 : input[i];
    lower[i] = '\0';
    if (exec_out) *exec_out = 0;
    if (match_idx) *match_idx = -1;

    for (int j = 0; j < lookdown_count; j++) {
        if (lookdown_table[j].flags & LOOKDOWN_EXEC) continue;
        if (fnmatch(lookdown_table[j].pattern, lower, 0) == 0) {
            if (match_idx) *match_idx = j;
            FILE *fp = popen(lookdown_table[j].command, "r");
            if (!fp) return -1;
            int len = fread(output, 1, output_max - 1, fp);
            output[len] = '\0';
            pclose(fp);
            while (len > 0 && output[len - 1] == '\n') output[--len] = '\0';
            return len;
        }
    }
    return -1;
}

static void hosuni_exec(const char *command) {
    platform_yield();

    sigset_t block, old;
    sigemptyset(&block);
    sigaddset(&block, SIGTERM);
    sigaddset(&block, SIGINT);
    sigprocmask(SIG_BLOCK, &block, &old);

    pid_t pid = fork();
    if (pid == 0) {
        sigprocmask(SIG_SETMASK, &old, NULL);
        platform_close_child_fds();
        execlp("sh", "sh", "-c", command, NULL);
        _exit(127);
    }
    if (pid > 0)
        waitpid(pid, NULL, 0);

    sigprocmask(SIG_SETMASK, &old, NULL);
    platform_reclaim();
    platform_drain_input();
}

## How it renders

hosuni.c talks directly to the display hardware:

  Linux:  DRM/KMS — opens /dev/dri/card*, sets mode, double-buffered
          page flip with vsync. Reads keyboard via /dev/input/event*.

  macOS:  Cocoa — NSWindow with a backing bitmap, keyboard events via
          NSEvent. Same rendering code, different platform layer.

Both platforms implement the contract defined in hosuni.h:
platform_init, platform_event, platform_flip, platform_cleanup
(plus platform_yield and platform_reclaim for child process handoff).
The rendering code (text, rectangles, scrolling) is shared.

Font rendering uses stb_truetype (single-header, vendored).

## Inference

hosuni.c builds its own HTTP requests over raw TCP sockets. No curl,
no HTTP library. It speaks the OpenAI-compatible chat completions
format:

  - Streaming (SSE) for READ mode — tokens render as they arrive
  - Non-streaming with tool calls — the model can call web_search,
    web_fetch, run_command, read_file, and list_dir. hosuni.c
    executes them and feeds results back in a multi-turn loop.
  - Thinking tag filter strips Gemma-4 <|channel>...</channel|> blocks

hosuni-discord (runtime/hosuni-discord.sh) wraps the same inference
path for delivery over Discord.

## Build

  Linux:  cc -O2 -o hosuni hosuni.c platform_linux.c -ldrm
  macOS:  cc -O2 -o hosuni hosuni.c platform_macos.m -framework Cocoa
