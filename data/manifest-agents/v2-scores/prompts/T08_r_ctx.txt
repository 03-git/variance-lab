   json escape (shared — every inference call site uses this)

static int json_escape(const char *in, char *out, int out_max) {
    int o = 0;
    for (int i = 0; in[i] && o < out_max - 6; i++) {
        switch (in[i]) {
        case '"':  out[o++] = '\\'; out[o++] = '"'; break;
        case '\\': out[o++] = '\\'; out[o++] = '\\'; break;
        case '\n': out[o++] = '\\'; out[o++] = 'n'; break;
        case '\r': out[o++] = '\\'; out[o++] = 'r'; break;
        case '\t': out[o++] = '\\'; out[o++] = 't'; break;
        default:
            if ((unsigned char)in[i] < 0x20) {
                o += snprintf(out + o, out_max - o, "\\u%04x", in[i]);
            } else {
                out[o++] = in[i];
            }
        }
    }
    out[o] = 0;
    return o;
}

    FILE *fp = fopen(path, "r");

    int line_num = 0;
    char line[LOOKDOWN_PAT_MAX + LOOKDOWN_CMD_MAX + 8];
    while (fgets(line, sizeof(line), fp) && lookdown_count < MAX_LOOKDOWN) {
        line_num++;
        int len = strlen(line);
        if (len > 0 && line[len - 1] == '\n') line[--len] = '\0';
        if (len == 0 || line[0] == '#') continue;
        char *tab = strchr(line, '\t');
        if (!tab) continue;
        *tab = '\0';

        uint8_t flags = 0;

            char *close = strchr(cmd, ']');

                    (strncmp(cmd, "[player]", 8) == 0 || strncmp(cmd, "[exec]", 6) == 0))
                    flags = LOOKDOWN_EXEC;
                cmd = tab2 + 1;
            } else continue;
        }
        if (!*cmd) continue;

static int lookdown_check(const char *input, char *output, int output_max,
                          int *exec_out, int *match_idx) {
    char lower[4096];
    int i;
    for (i = 0; input[i] && i < (int)sizeof(lower) - 1; i++)
        lower[i] = (input[i] >= 'A' && input[i] <= 'Z') ? input[i] + 32 : input[i];
    lower[i] = '\0';
    if (exec_out) *exec_out = 0;
    if (match_idx) *match_idx = -1;

    for (int j = 0; j < lookdown_count; j++) {
        if (lookdown_table[j].flags & LOOKDOWN_EXEC) continue;
        if (fnmatch(lookdown_table[j].pattern, lower, 0) == 0) {
            if (match_idx) *match_idx = j;
            FILE *fp = popen(lookdown_table[j].command, "r");
            if (!fp) return -1;
            int len = fread(output, 1, output_max - 1, fp);
            output[len] = '\0';
            pclose(fp);
            while (len > 0 && output[len - 1] == '\n') output[--len] = '\0';
            return len;
        }
    }
    return -1;
}
