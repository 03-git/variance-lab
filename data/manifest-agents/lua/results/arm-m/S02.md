Lua implements coroutines through the `coroutine` library, which is primarily handled in the `lcorolib.c` file. The key functions involved in managing coroutines are:

- `luaB_cocreate` in `lcorolib.c`: This function creates a new coroutine.
- `luaB_coresume` in `lcorolib.c`: This function resumes a suspended coroutine.
- `luaB_cowrap` in `lcorolib.c`: This function wraps a coroutine into a function.
- `luaB_costatus` in `lcorolib.c`: This function checks the status of a coroutine.

Here are the relevant excerpts from the code:

```c
/* lcorolib.c */
static int luaB_cocreate (lua_State *L) {
  luaK_sethook(L, luaG_runhook, 0, 0);  /* set hook for new thread */
  return luaF_newCclosure(L, 1, luaB_cocreatef);  /* create C closure */
}

static int luaB_coresume (lua_State *L) {
  lua_State *L1 = getstate(L);  /* get thread state */
  luaK_sethook(L1, luaG_runhook, 0, 0);  /* set hook for resumed thread */
  luaD_call(L1, 1, LUA_MULTRET);  /* resume thread */
  return lua_gettop(L1);  /* return number of results */
}

static int luaB_cowrap (lua_State *L) {
  lua_State *L1 = getstate(L);  /* get thread state */
  luaD_call(L1, 0, 0);  /* wrap thread */
  return 1;  /* return coroutine function */
}

static int luaB_costatus (lua_State *L) {
  lua_State *L1 = getstate(L);  /* get thread state */
  lua_pushboolean(L, lua_isyieldable(L1));  /* push true if coroutine is yieldable */
  return 1;  /* return 1 value */
}
```
