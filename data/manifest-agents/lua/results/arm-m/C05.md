static void luaG_typeerror (lua_State *L, const char *op, TValue *v) {
  lua_pushliteral(L, "type");
  lua_pushstring(L, lua_typename(L, TYPE(v)));
  lua_pushliteral(L, op);
  lua_concat(L, 3);
  lua_error(L);
}

static void luaG_getfuncline (lua_State *L, const Proto *f, int i) {
  int line = 0;
  if (f->linedefined <= i && i <= f->lastlinedefined) {
    line = f->source[i - f->linedefined];
  }
  lua_pushinteger(L, line);
}
