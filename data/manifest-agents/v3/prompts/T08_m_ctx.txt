static int json_escape(const char *in, char *out, int out_max) {
    int o = 0;
    for (int i = 0; in[i] && o < out_max - 6; i++) {
        switch (in[i]) {
        case '"':  out[o++] = '\\'; out[o++] = '"'; break;
        case '\\': out[o++] = '\\'; out[o++] = '\\'; break;
        case '\n': out[o++] = '\\'; out[o++] = 'n'; break;
        case '\r': out[o++] = '\\'; out[o++] = 'r'; break;
        case '\t': out[o++] = '\\'; out[o++] = 't'; break;
        default:
            if ((unsigned char)in[i] < 0x20) {
                o += snprintf(out + o, out_max - o, "\\u%04x", in[i]);
            } else {
                out[o++] = in[i];
            }
        }
    }
    out[o] = 0;
    return o;
}

## ./runtime/bin/lookdown-audit.sh (lines 1-40 of 40)
#!/bin/bash
# lookdown-audit.sh — shadow detection using the same [[ ]] glob the handler uses
set -euo pipefail

NODE="${1:-$HOME/.subtract/lookdown.$(hostname -s).tsv}"
UNIVERSAL="${2:-$HOME/.subtract/lookdown.universal.tsv}"

# load in handler order: node first, then universal
entries=()
files=()
for f in "$NODE" "$UNIVERSAL"; do
    [ -f "$f" ] || continue
    n=0
    while IFS= read -r line; do
        n=$((n + 1))
        [[ -z "$line" || "$line" == \#* ]] && continue
        pat="${line%%	*}"
        entries+=("$pat")
        files+=("$(basename "$f"):$n")
    done < "$f"
done

total=${#entries[@]}
shadows=0

for ((i=0; i<total; i++)); do
    pat_i="${entries[$i]}"
    cand="${pat_i//\*/x}"
    cand=$(printf '%s' "$cand" | tr '[:upper:]' '[:lower:]')
    for ((j=0; j<i; j++)); do
        pat_j="${entries[$j]}"
        if [[ "$cand" == $pat_j ]]; then
            shadows=$((shadows + 1))
            printf 'shadow: %s %s eaten by %s %s\n' "${files[$i]}" "$pat_i" "${files[$j]}" "$pat_j"
            break
        fi
    done
done

printf '\n%d entries, %d shadows\n' "$total" "$shadows"

  Linux:  DRM/KMS — opens /dev/dri/card*, sets mode, double-buffered

# hosuni.c

Framebuffer computing for humans. ~2,200 lines of C, no dependencies
beyond libc, libdrm (Linux), and Cocoa (macOS). Compiles to a single
binary under 200KB.

hosuni.c is the subtract.ing interface that replaced the browser-based
demos (demo.html, multiplying.html, the clinical pages). Those pages
required a Python bridge, a local HTTP server, and a browser. hosuni.c
needs a keyboard and a screen.

Named for the digital governess in the subtract.ing formation —
the process that sits with the kids at the framebuffer while the
governor works.

## What it does

Three modes, plus a starred-launch shortcut:

  READ  (F1)  Ask a question, get a streamed answer from the local
              inference server. Conversation history persists across
              turns. Shows time-to-first-token and tokens/sec.

  PLAY  (F2)  Launch the starred entry from the lookdown table
              directly. One keypress to resume whatever was last
              starred in RUN mode.

  RUN   (F3)  Type-to-filter launcher over the lookdown table.
              Every [player] entry in lookdown.tsv becomes a
              launchable program. Star entries for quick access
              via F2.

  TEACH (F4)  vi-subset text editor. Gap buffer, normal/insert/command
              modes, yank/paste, line numbers, ex commands (:w, :q).

White text on black background, fixed-width font rendered via
stb_truetype. The cursor is a colored block. The status bar shows
the current mode, and in READ mode, inference timing.
