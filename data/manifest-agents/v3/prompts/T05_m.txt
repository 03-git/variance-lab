/*
 * syna_poll — i2c-hid polling for GPIO-locked touchpads
 *
 * When BIOS locks the interrupt pin in native mode, the i2c-hid
 * device registers but never fires interrupts. This module polls
 * the HID input report register on a timer and injects the data
 * into the existing i2c-hid pipeline.
 *
 * Build:
 *   make -C /lib/modules/$(uname -r)/build M=$(pwd) modules
 *
 * Usage:
 *   insmod syna_poll.ko [poll_ms=10] [vendor=0x06cb] [product=0xcd41]
 */

#include <linux/module.h>
#include <linux/i2c.h>
#include <linux/hid.h>
#include <linux/input.h>
#include <linux/timer.h>
#include <linux/workqueue.h>
#include <linux/slab.h>
#include <linux/atomic.h>
#include <linux/delay.h>

#define DEFAULT_POLL_MS  10
#define MIN_POLL_MS       1
#define MAX_REPORT_SIZE  64
#define HID_DESC_REG     0x0020

static int poll_ms = DEFAULT_POLL_MS;
module_param(poll_ms, int, 0444);
MODULE_PARM_DESC(poll_ms, "Polling interval in milliseconds (default 10)");

static int vendor = 0x06CB;
module_param(vendor, int, 0444);
MODULE_PARM_DESC(vendor, "I2C HID vendor ID");

static int product = 0xCD41;
module_param(product, int, 0444);

/*
 * parsimony — inference as a unix filter
 *
 * Loads a GGUF once. Reads JSON requests from stdin (one per line).
 * Writes JSON responses to stdout (one per line). Stays alive.
 * The shell manages the lifecycle via coproc. Stateless: KV cache
 * clears between requests; the orchestrator owns conversation state.
 *
 * Wire format (input, one JSON line per request):
 *   {"messages":[{"role":"...","content":"..."},...],"max_tokens":N}
 *   OpenAI chat-completions-shaped. UTF-8 bytes, no \uXXXX escapes.
 *   No nested objects in content, no JSON inside JSON. Caller
 *   flattens with jq -c if needed (jq emits UTF-8 by default).
 *   Input is read dynamically via getline (no fixed cap).
 *
 * Wire format (output, one JSON line per response):
 *   {"choices":[{"message":{"role":"assistant","content":"..."},"finish_reason":"stop"}]}
 *
 * Wire format (output, streaming — when input has "stream":true):
 *   data: {"choices":[{"delta":{"content":"token"},"finish_reason":null}]}
 *   data: {"choices":[{"delta":{"content":"token"},"finish_reason":null}]}
 *   ...
 *   data: {"choices":[{"delta":{},"finish_reason":"stop"}]}
 *   data: [DONE]
 *
 * Build:
 *   cc -o parsimony parsimony.c -I/path/to/llama.cpp/include \
 *      -L/path/to/llama.cpp/lib -lllama -lggml -lm -lpthread
 *
 * Usage:
 *   coproc INF { ./parsimony model.gguf; }
 *   echo '{"messages":[{"role":"user","content":"hello"}]}' >&${INF[1]}
 *   read -r response <&${INF[0]}
 *
 *   # with standing context:
 *   coproc INF { ./parsimony model.gguf --prelude-file ~/subtract.ing/SOUL.txt; }
